116 	/**
117 	 * Prepares the URL and sends a request to BioMart.
118 	 * 
119 	 * @param baseUrl the BioMart base URL.
120 	 * @param query the query (XML) sent to BioMart.
121 	 * @return a URLConnection.
122 	 * @throws BiomartFetchDataException
123 	 *             in case of problem with the URL, its encoding or while
124 	 *             opening the connection.
125 	 */
126 	protected URLConnection sendRequest(String baseUrl, String query)
127 	throws BiomartFetchDataException {
128 		URL url = null;
129 		try {
130 			url = DataTypeConverter.createEncodedUrl(baseUrl, query);
131 		} catch (UnsupportedEncodingException ex) {
132 			throw new BiomartFetchDataException(
133 					"Failed to encode the url for Biomart request " + baseUrl
134 					+ " " + query, ex);
135 		} catch (MalformedURLException ex) {
136 			throw new BiomartFetchDataException(
137 					"Failed to create the url for Biomart request " + baseUrl
138 					+ " " + query, ex);
139 		}
140 		URLConnection uCon = null;
141 		try {
142 			uCon = url.openConnection();
143 		} catch (IOException ex) {
144 			throw new BiomartFetchDataException(
145 					"Unable to connect to Biomart service to process request "
146 					+ url.toString(), ex);
147 		}
148 		return uCon;
149 	}

151    /**
152     * Parses the response from BioMart containing pathways info.
153     * @param uCon
154     * @return a list of Reactome pathway stable IDs.
155     * @throws BiomartFetchDataException in case of error reading from the
156     * 		connection, or if BioMart returns an error message.
157     */
158    private static List<String> parsePathwaysResponse(URLConnection uCon) 
159    throws BiomartFetchDataException {
160        List<String> results = new ArrayList<String>();
161         BufferedReader in = null;
162         try {
163             in = new BufferedReader(new InputStreamReader(uCon.getInputStream()));
164             String inputLine, errorMsg = "";
165             while ((inputLine = in.readLine()) != null) {
166             	if (inputLine.indexOf("ERROR") > -1){
167             		errorMsg += inputLine;
168             	} else {
169                     final String pathwayId = inputLine.trim();
170                     if (!results.contains(pathwayId)){
171     					results.add(pathwayId);
172                     }
173             	}
174             }
175             if (errorMsg.length() > 0){
176             	LOGGER.error(errorMsg);
177         		throw new BiomartFetchDataException(errorMsg);
178         	}
179         } catch (IOException ex) {
180             throw new BiomartFetchDataException("Failed to read the response of the Url connection"
181             		+ uCon.toString(), ex);
182         } finally {
183             try {
184 				in.close();
185 			} catch (IOException e) {
186 				LOGGER.error("Unable to close BufferedReader", e);
187 			}
188         }
189         LOGGER.debug("Pathways parsed: " + results);
190        return results;
191    }

193 	/**
194 	* Parses the response from BioMart containing reaction info.
195 	* @param uCon a connection to BioMart.
196 	* @return a list of ReactionPathway objects containing information only
197 	* 		about reactions, whose xrefs will be Reactome reaction IDs (not
198 	* 		whole URLs).
199 	* @throws BiomartFetchDataException in case of problem reading from the
200 	* 		connection.
201 	*/
202    private static List<ReactionPathway> parseReactionsResponse(URLConnection uCon)
203    throws BiomartFetchDataException {
204        List<ReactionPathway> reactionPathways = new ArrayList<ReactionPathway>();
205         BufferedReader in = null;
206         try {
207             in = new BufferedReader(new InputStreamReader(uCon.getInputStream()));
208             String inputLine;
209             while ((inputLine = in.readLine()) != null) {
210                 String[]fieldValues = inputLine.split("\t");
211                 ReactionPathway reactionPathway = new ReactionPathway();
212                 EnzymeReaction enzymeReaction = new EnzymeReaction();
213                 String reactionId = fieldValues[0].trim();
214                 List<Object> reactomeReactionId = new ArrayList<Object>();
215                 reactomeReactionId.add(reactionId);
216                 enzymeReaction.setXrefs(reactomeReactionId);
217                 String reactionName = fieldValues[1].trim();                
218                 enzymeReaction.setName(reactionName);
219                 reactionPathway.setReaction(enzymeReaction);
220                 reactionPathways.add(reactionPathway);
221             }
222             in.close();
223         } catch (IOException ex) {
224             throw new BiomartFetchDataException("Failed to read the response of the Url connection" +uCon.toString(), ex);
225         }
226        return reactionPathways;
227    }
