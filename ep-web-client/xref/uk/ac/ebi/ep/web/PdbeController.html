27      @RequestMapping(value = "/ajax/pdbe/{pdbId}")
28      protected String getStructure(Model model, @PathVariable String pdbId) {
29          String retValue = "proteinStructure-single";
30          ProteinStructure structure = null;
31          try {
32              SimpleDASFeaturesAdapter pdbeAdapter =
33                      new SimpleDASFeaturesAdapter(IDASFeaturesAdapter.PDBE_DAS_URL);
34              SegmentAdapter segment = pdbeAdapter.getSegment(pdbId);
35              structure = new ProteinStructure();
36              structure.setId(segment.getId());
37              for (FeatureAdapter feature : segment.getFeature()) {
38                  if (feature.getType().getId().equals("description")) {
39                      structure.setDescription(feature.getNotes().get(0)); // FIXME?
40                  } else if (feature.getType().getId().equals("image")) {
41                      Image image = new Image();
42                      image.setSource(feature.getLinks().get(0).getHref());
43                      image.setCaption(feature.getLinks().get(0).getContent());
44                      image.setHref(feature.getLinks().get(1).getHref());
45                      structure.setImage(image);
46                  } else if (feature.getType().getId().equals("provenance")) {
47                      structure.setProvenance(feature.getNotes());
48                  } else if (feature.getType().getId().equals("summary")) {
49                      DASSummary summary = new DASSummary();
50                      summary.setLabel(feature.getLabel());
51                      summary.setNote(feature.getNotes());
52                      structure.getSummary().add(summary);
53                  }
54              }
55      		model.addAttribute("proteinStructure", structure);
56  		} catch (Exception e) {
57  			LOGGER.error("Unable to retrieve structure " + pdbId, e);
58  			retValue = "error";
59  		}
60  
61  		return retValue;
62  	}
