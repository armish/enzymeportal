8   /**
9    * The Severity class is used to specify the severity of an exception that has occurred
10   * within the project. This is used to allow the catching component to act upon the caught
11   * exception in the most appropriate manner.
12   * @author Joseph
13   */
14  public enum Severity {
15      /**
16       * WARNING level indicates that this exception should be logged, but that it has
17       * no adverse effect on the operation of the system in any respect.
18       */
19      WARNING,
20      /**
21       * TASK AFFECTING exceptions indicate that the current task will be unable to
22       * complete, but that the project's operation is otherwise unaffected.
23       */
24      TASK_AFFECTING,
25      /**
26       * COMPONENT AFFECTING exceptions indicate that the current component will not
27       * continue functioning in its current state, but that the system can be
28       * recovered without requiring a software or hardware re-initialisation.
29       */
30      COMPONENT_AFFECTING,
31      /**
32       * SYSTEM AFFECTING exceptions indicate that the Application is unable to continue
33       * functioning. To rectify the problem, a software restart or complete hardware
34       * shutdown/reboot may be required.
35       */
36      SYSTEM_AFFECTING,
37      /**
38       * EXTERNAL SYSTEM AFFECTING exceptions indicate that an external system is
39       * adversely affected by the exception encountered. This is the most critical
40       * of all reportable severities, as it may require manual hardware or software
41       * re-initialisation of those external systems.
42       */
43      EXTERNAL_SYSTEM_AFFECTING
44  }
