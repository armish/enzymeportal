55       * this method takes an input to generate a siteMap

57       * @param inputData collection of data to be used in generating the siteMap

62       * @throws EnzymePortalException while generating or writing the siteMap
63       */
64      public void generateSitemap(Collection<?> inputData, File output, String filename_prefix, boolean testMode) throws SiteMapException {
65          WebSitemapGenerator sitemapGenerator;
66          WebSitemapUrl url = null;
67          //SitemapIndexGenerator indexGenerator;
68  
69          // Use DAY pattern (2012-07-13), Greenwich Mean Time timezone
70          W3CDateFormat dateFormat = new W3CDateFormat(Pattern.DAY);
71          dateFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
72  
73          try {
74              //indexGenerator = new SitemapIndexGenerator(SITEMAP_LOCATION, output);
75              //sitemapGenerator = WebSitemapGenerator.builder(SITEMAP_LOCATION, output).allowMultipleSitemaps(true).fileNamePrefix(filename_prefix).gzip(false).dateFormat(dateFormat).build();
76               sitemapGenerator = WebSitemapGenerator.builder(SITEMAP_LOCATION, output).fileNamePrefix(filename_prefix).gzip(false).dateFormat(dateFormat).build();
77              for (Object input : inputData) {
78  
79                  for (String entry_tab : tabs) {
80  
81                      url = new WebSitemapUrl.Options(MessageFormat.format(entry_tab, new Object[]{input})).lastMod(new Date()).priority(1.0).changeFreq(ChangeFreq.HOURLY).build();
82                      sitemapGenerator.addUrl(url);
83                  }
84  
85  
86  
87              }
88              sitemapGenerator.write();
89              sitemapGenerator.writeSitemapsWithIndex(SITEMAP_INDEX);
90  
91              if (testMode) {
92                  sitemapGenerator.deleteGeneratedSiteMap();
93              }
94  
95          } catch (MalformedURLException ex) {
96              LOGGER.fatal("MalformedURLException ", ex);
97          }
98  
99  
100 
101     }
