81      /**
82       * this method takes an input to generate a siteMap
83       * @param inputData collection of data to be used in generating the siteMap
84       * @param output  the siteMap generated
85       * @throws EnzymePortalException while generating or writing the siteMap
86       */
87      public void generateSitemap(Collection<?> inputData, File output,String name,boolean testMode) throws SiteMapException {
88  
89          Urlset urlset = objectFactory.createUrlset();
90          Url enzymePortalUrl = objectFactory.createUrl();
91          enzymePortalUrl.setLoc(ENZYME_PORTAL_URL);
92          enzymePortalUrl.setLastmod(new SimpleDateFormat("yyyy-MM-dd").format(new Date()));
93          enzymePortalUrl.setChangefreq(CHANGE_FREQ);
94          enzymePortalUrl.setPriority(new BigDecimal("0.8").setScale(1, BigDecimal.ROUND_DOWN));
95          urlset.getUrl().add(enzymePortalUrl);
96         
97          for (Object input : inputData) {
98              for (String entry_tab : tabs) {
99                  Url pageUrl = objectFactory.createUrl();
100                 pageUrl.setLoc(MessageFormat.format(entry_tab,
101                         new Object[]{input}));
102          
103                 pageUrl.setChangefreq(CHANGE_FREQ);
104                 urlset.getUrl().add(pageUrl);
105           }
106         }
107         try {
108             marshaller.marshal(urlset, output);
109             makeGzip(output);
110         } catch (Exception ex) {
111             throw new SiteMapException("JAXBException while marshalling the output", ex, Severity.SYSTEM_AFFECTING);
112         }
113     }
