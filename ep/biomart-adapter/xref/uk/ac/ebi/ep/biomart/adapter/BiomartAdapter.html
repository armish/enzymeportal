97  	/**
98  	 * Prepares the URL and sends a request to BioMart.
99  	 * 
100 	 * @param baseUrl the BioMart base URL.
101 	 * @param query the query (XML) sent to BioMart.
102 	 * @return a URLConnection.
103 	 * @throws BiomartFetchDataException
104 	 *             in case of problem with the URL, its encoding or while
105 	 *             opening the connection.
106 	 */
107 	protected URLConnection sendRequest(String baseUrl, String query)
108 	throws BiomartFetchDataException {
109 		URL url = null;
110 		try {
111 			url = DataTypeConverter.createEncodedUrl(baseUrl, query);
112 		} catch (UnsupportedEncodingException ex) {
113 			throw new BiomartFetchDataException(
114 					"Failed to encode the url for Biomart request " + baseUrl
115 					+ " " + query, ex);
116 		} catch (MalformedURLException ex) {
117 			throw new BiomartFetchDataException(
118 					"Failed to create the url for Biomart request " + baseUrl
119 					+ " " + query, ex);
120 		}
121 		URLConnection uCon = null;
122 		try {
123 			uCon = url.openConnection();
124 		} catch (IOException ex) {
125 			throw new BiomartFetchDataException(
126 					"Unable to connect to Biomart service to process request "
127 					+ url.toString(), ex);
128 		}
129 		return uCon;
130 	}

132    /**
133     * Parses the response from BioMart containing pathways info.
134     * @param uCon
135     * @return a list of Reactome pathway stable IDs.
136     * @throws BiomartFetchDataException in case of error reading from the
137     * 		connection, or if BioMart returns an error message.
138     */
139    private static List<String> parsePathwaysResponse(URLConnection uCon) 
140    throws BiomartFetchDataException {
141        List<String> results = new ArrayList<String>();
142         BufferedReader in = null;
143         try {
144             in = new BufferedReader(new InputStreamReader(uCon.getInputStream()));
145             String inputLine, errorMsg = "";
146             while ((inputLine = in.readLine()) != null) {
147             	if (inputLine.indexOf("ERROR") > -1){
148             		errorMsg += inputLine;
149             	} else {
150                     final String pathwayId = inputLine.trim();
151                     if (!results.contains(pathwayId)){
152     					results.add(pathwayId);
153                     }
154             	}
155             }
156             if (errorMsg.length() > 0){
157             	LOGGER.error(errorMsg);
158         		throw new BiomartFetchDataException(errorMsg);
159         	}
160         } catch (IOException ex) {
161             throw new BiomartFetchDataException("Failed to read the response of the Url connection"
162             		+ uCon.toString(), ex);
163         } finally {
164             try {
165 				in.close();
166 			} catch (IOException e) {
167 				LOGGER.error("Unable to close BufferedReader", e);
168 			}
169         }
170         LOGGER.debug("Pathways parsed: " + results);
171        return results;
172    }

174 	/**
175 	* Parses the response from BioMart containing reaction info.
176 	* @param uCon a connection to BioMart.
177 	* @return a list of ReactionPathway objects containing information only
178 	* 		about reactions, whose xrefs will be Reactome reaction IDs (not
179 	* 		whole URLs).
180 	* @throws BiomartFetchDataException in case of problem reading from the
181 	* 		connection.
182 	*/
183    private static List<ReactionPathway> parseReactionsResponse(URLConnection uCon)
184    throws BiomartFetchDataException {
185        List<ReactionPathway> reactionPathways = new ArrayList<ReactionPathway>();
186         BufferedReader in = null;
187         try {
188             in = new BufferedReader(new InputStreamReader(uCon.getInputStream()));
189             String inputLine;
190             while ((inputLine = in.readLine()) != null) {
191                 String[]fieldValues = inputLine.split("\t");
192                 if(fieldValues.length >1){
193                 ReactionPathway reactionPathway = new ReactionPathway();
194                 EnzymeReaction enzymeReaction = new EnzymeReaction();
195                 String reactionId = fieldValues[0].trim();
196                
197                 List<Object> reactomeReactionId = new ArrayList<Object>();
198                 reactomeReactionId.add(Database.REACTOME.getEntryUrl(reactionId));
199                 enzymeReaction.setXrefs(reactomeReactionId);
200                 
201                 
202                 String reactionName = fieldValues[1].trim();                
203                 enzymeReaction.setName(reactionName);
204                 
205                 reactionPathway.setReaction(enzymeReaction);
206                 reactionPathways.add(reactionPathway);
207                 }
208             }
209             in.close();
210         } catch (IOException ex) {
211             throw new BiomartFetchDataException("Failed to read the response of the Url connection" +uCon.toString(), ex);
212         }
213        return reactionPathways;
214    }
