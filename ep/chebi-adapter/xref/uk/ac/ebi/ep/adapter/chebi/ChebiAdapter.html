98      public Entity querySynonymsForChebiId(String name, List<String> ids) throws ChebiFetchDataException {
99          Entity chebiEntityMappedToSynonym = null;
100         Entity chebiEntityMappedToUniprotSynonym = null;
101         Entity chebiEntitySelected = null;
102         try {
103             List<Entity> results = client.getCompleteEntityByList(ids);
104             for (Entity result:results) {
105                 List<DataItem> synonyms = result.getSynonyms();
106                 // List all synonyms
107                 for (DataItem dataItem : synonyms) {                    
108                     String source = dataItem.getSource();
109                     String synonymName = dataItem.getData().trim();
110                     if (source.equalsIgnoreCase("UniProt")) {
111                         if (synonymName.equalsIgnoreCase(name)) {
112                             chebiEntityMappedToUniprotSynonym = result;
113                         }
114                     } //other synonyms different from Uniprot
115                     else {
116                         if (synonymName.equalsIgnoreCase(name)) {
117                             chebiEntityMappedToSynonym = result;
118                             break;
119                         }
120                     }
121                 }
122             if (chebiEntityMappedToUniprotSynonym != null) {
123                 chebiEntitySelected = chebiEntityMappedToUniprotSynonym;
124                 break;
125             } else {
126                 if (chebiEntityMappedToSynonym != null) {
127                     chebiEntitySelected = chebiEntityMappedToSynonym;
128                     break;
129                 }
130             }
131 
132             }
133         } catch (ChebiWebServiceFault_Exception e) {
134             throw new ChebiFetchDataException("Failed to get complete entity list for " +ids.toString(), e);
135         }
136 
137         return chebiEntitySelected;
138     }

175     public Map<String,Entity> queryChebiAllNamesForEntities(Set<String> names) throws ChebiFetchDataException  {
176         Map<String,Entity> nameEntityMap = new HashMap<String, Entity>();
177         for (String name:names) {
178             //1st choice: exact chebi name. 2nd choice: uniprot synonym. 3rd choice: other synonym
179             Entity chebiEntity = null;
180             String chebiId = queryChebiNameForId(name);
181             if (chebiId != null) {
182                 chebiEntity = getChebiCompleteEntity(chebiId);
183             } else {
184                 List<String> candidateIds = queryChebiAllNamesForIds(name,ChebiAdapter.IDS_MAX_SIZE);
185                 if (candidateIds.size() > 0) {
186                     chebiEntity = querySynonymsForChebiId(name, candidateIds);
187                 }
188             }
189             nameEntityMap.put(name, chebiEntity);
190         }
191         return nameEntityMap;
192     }

206     public EnzymeModel getMoleculeCompleteEntries(EnzymeModel enzymeModel) throws ChebiFetchDataException {
207         ChemicalEntity chemicalEntity = enzymeModel.getMolecule();
208         ExecutorService pool = Executors.newFixedThreadPool(config.getMaxThreads());
209 //	    ExecutorService pool = Executors.newCachedThreadPool();
210 	    CompletionService<Molecule> ecs =
211 	    		new ExecutorCompletionService<Molecule>(pool);
212 
213         final CountableMolecules drugs = chemicalEntity.getDrugs();
214         final CountableMolecules inhibitors = chemicalEntity.getInhibitors();
215         final CountableMolecules activators = chemicalEntity.getActivators();
216         final CountableMolecules cofactors = chemicalEntity.getCofactors();
217         final CountableMolecules bioactive =
218                 chemicalEntity.getBioactiveLigands();
219 	    try {
220         	LOGGER.debug("MOLECULES before submitting drugs");
221             Map<Future<Molecule>, Molecule> drugFut =
222 	    			getFuture2MoleculeMap(drugs, ecs);
223         	LOGGER.debug("MOLECULES before submitting inhibitors");
224             Map<Future<Molecule>, Molecule> inhFut =
225 		    		getFuture2MoleculeMap(inhibitors, ecs);
226         	LOGGER.debug("MOLECULES before submitting activators");
227             Map<Future<Molecule>, Molecule> actFut =
228 	        		getFuture2MoleculeMap(activators, ecs);
229         	LOGGER.debug("MOLECULES before submitting cofactors");
230             Map<Future<Molecule>, Molecule> cofFut =
231 	        		getFuture2MoleculeMap(cofactors, ecs);
232         	LOGGER.debug("MOLECULES before submitting bioactives");
233             Map<Future<Molecule>, Molecule> bioactFut =
234 	        		getFuture2MoleculeMap(bioactive, ecs);
235 	        // How many jobs have we sent?
236 	        final int numOfMolecules = drugFut.size() + inhFut.size()
237 	        		+ actFut.size() + cofFut.size() + bioactFut.size();
238 	        // Let's retrieve them:
239         	LOGGER.debug("MOLECULES before getting molecules");
240 			for (int i = 0; i < numOfMolecules; i++){
241 	        	Future<Molecule> future = null;
242 	        	try {
243 	        		future = ecs.take();
244 //	        		future = ecs.poll(config.getTimeout(), TimeUnit.MILLISECONDS);
245 	        		if (future != null){
246 	        			final Molecule molecule = future.get();
247 	        			if (molecule != null){
248 		        			// Replace the incomplete molecule with the complete one:
249 	        				if (drugFut.containsKey(future)){
250 	        					drugFut.put(future, molecule);
251 	        				} else if (inhFut.containsKey(future)){
252 	        					inhFut.put(future, molecule);
253 	        				} else if (actFut.containsKey(future)){
254 	        					actFut.put(future, molecule);
255 	        				} else if (cofFut.containsKey(future)){
256 	        					cofFut.put(future, molecule);
257 	        				} else if (bioactFut.containsKey(future)){
258 	        					bioactFut.put(future, molecule);
259 	        				}
260 	        			} else {
261 		    				LOGGER.warn("Molecule retrieved is null!");
262 	        			}
263 	        		} else {
264 	    				LOGGER.warn("Job result not retrieved!");
265 	        		}
266 	        	} catch (Exception e){
267 	            	// Don't stop the others
268 	        		LOGGER.error("Callable " + (i+1) + " of " + numOfMolecules
269 	            			+ " - " + e.getMessage(), e);
270 	        	}
271 	        }
272         	LOGGER.debug("MOLECULES before setting molecules");
273         	// Replace first items in each list with the retrieved ones:
274         	if (drugs != null){
275                 drugs.getMolecule().subList(0, drugFut.size()).clear();
276                 drugs.getMolecule().addAll(0, drugFut.values());
277             }
278             if (inhibitors != null){
279                 inhibitors.getMolecule().subList(0, inhFut.size()).clear();
280                 inhibitors.getMolecule().addAll(0, inhFut.values());
281             }
282             if (activators != null){
283                 activators.getMolecule().subList(0, actFut.size()).clear();
284                 activators.getMolecule().addAll(0, actFut.values());
285             }
286             if (cofactors != null){
287                 cofactors.getMolecule().subList(0, cofFut.size()).clear();
288                 cofactors.getMolecule().addAll(0, cofFut.values());
289             }
290             if (bioactive != null){
291                 bioactive.getMolecule().subList(0, bioactFut.size()).clear();
292                 bioactive.getMolecule().addAll(0, bioactFut.values());
293             }
294 	    } finally {
295 	    	pool.shutdown();
296 	    }
297         return enzymeModel;
298     }

300     /**
301      * Prepares callables and sends them to a completion service in order to
302      * retrieve complete entities from ChEBI.
303      * @param molecules the list of molecules to be retrieved from ChEBI.
304      * @param ecs the completion service where the jobs are submitted.
305      * @return a map of <code>Future</code>s to <code>Molecule</code>s
306      */
307     private Map<Future<Molecule>, Molecule> getFuture2MoleculeMap(
308     		CountableMolecules molecules, CompletionService<Molecule> ecs){
309     	Map<Future<Molecule>, Molecule> fut2mol =
310     			new LinkedHashMap<Future<Molecule>, Molecule>();
311         if (molecules != null && molecules.getMolecule() != null){
312             for (Molecule molecule : molecules.getMolecule()) {
313                 ChebiWsCallable callable = getCallable(molecule);
314                 if (callable != null){
315                     fut2mol.put(ecs.submit(callable), molecule);
316                 }
317                 // Limit to the first few molecules:
318                 if (fut2mol.size() >= config.getMaxRetrievedMolecules()) break;
319             }
320         }
321         return fut2mol;
322     }

325 	 * Prepares a callable suitable to retrieve a complete Molecule from ChEBI
326 	 * depending on the information available.
327 	 * @param molecule an incomplete Molecule.
328 	 * @return a callable to retrieve the complete Molecule.
329 	 */
330 	private ChebiWsCallable getCallable(Molecule molecule) {
331 		LOGGER.debug("before getting ChEBI callable");
332 		ChebiWsCallable callable = null;
333 		final String drugId = molecule.getId();
334 		if (drugId != null){
335 			callable = drugId.startsWith("CHEBI:")?
336 					new ChebiWsCallable(config, drugId,
337 							SearchCategory.CHEBI_ID):
338 					new ChebiWsCallable(config, drugId,
339 							SearchCategory.DATABASE_LINK_REGISTRY_NUMBER_CITATION);
340 		} else if (molecule.getName() != null){
341 			callable = new ChebiWsCallable(config, molecule.getName(),
342 					SearchCategory.ALL_NAMES);
343 		}
344 		if (callable == null){
345 			LOGGER.warn("The molecule has no id or name");
346 		}
347 		LOGGER.debug("after getting ChEBI callable");
348 		return callable;
349 	}
