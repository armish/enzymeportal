97      public Entity querySynonymsForChebiId(String name, List<String> ids) throws ChebiFetchDataException {
98          Entity chebiEntityMappedToSynonym = null;
99          Entity chebiEntityMappedToUniprotSynonym = null;
100         Entity chebiEntitySelected = null;
101         try {
102             List<Entity> results = client.getCompleteEntityByList(ids);
103             for (Entity result:results) {
104                 List<DataItem> synonyms = result.getSynonyms();
105                 // List all synonyms
106                 for (DataItem dataItem : synonyms) {                    
107                     String source = dataItem.getSource();
108                     String synonymName = dataItem.getData().trim();
109                     if (source.equalsIgnoreCase("UniProt")) {
110                         if (synonymName.equalsIgnoreCase(name)) {
111                             chebiEntityMappedToUniprotSynonym = result;
112                         }
113                     } //other synonyms different from Uniprot
114                     else {
115                         if (synonymName.equalsIgnoreCase(name)) {
116                             chebiEntityMappedToSynonym = result;
117                             break;
118                         }
119                     }
120                 }
121             if (chebiEntityMappedToUniprotSynonym != null) {
122                 chebiEntitySelected = chebiEntityMappedToUniprotSynonym;
123                 break;
124             } else {
125                 if (chebiEntityMappedToSynonym != null) {
126                     chebiEntitySelected = chebiEntityMappedToSynonym;
127                     break;
128                 }
129             }
130 
131             }
132         } catch (ChebiWebServiceFault_Exception e) {
133             throw new ChebiFetchDataException("Failed to get complete entity list for " +ids.toString(), e);
134         }
135 
136         return chebiEntitySelected;
137     }

174     public Map<String,Entity> queryChebiAllNamesForEntities(Set<String> names) throws ChebiFetchDataException  {
175         Map<String,Entity> nameEntityMap = new HashMap<String, Entity>();
176         for (String name:names) {
177             //1st choice: exact chebi name. 2nd choice: uniprot synonym. 3rd choice: other synonym
178             Entity chebiEntity = null;
179             String chebiId = queryChebiNameForId(name);
180             if (chebiId != null) {
181                 chebiEntity = getChebiCompleteEntity(chebiId);
182             } else {
183                 List<String> candidateIds = queryChebiAllNamesForIds(name,ChebiAdapter.IDS_MAX_SIZE);
184                 if (candidateIds.size() > 0) {
185                     chebiEntity = querySynonymsForChebiId(name, candidateIds);
186                 }
187             }
188             nameEntityMap.put(name, chebiEntity);
189         }
190         return nameEntityMap;
191     }

205     public EnzymeModel getMoleculeCompleteEntries(EnzymeModel enzymeModel) throws ChebiFetchDataException {
206         ChemicalEntity chemicalEntity = enzymeModel.getMolecule();
207         ExecutorService pool = Executors.newFixedThreadPool(config.getMaxThreads());
208 //	    ExecutorService pool = Executors.newCachedThreadPool();
209 	    CompletionService<Molecule> ecs =
210 	    		new ExecutorCompletionService<Molecule>(pool);
211 	    
212 	    try {
213         	LOGGER.debug("MOLECULES before submitting drugs");
214 	    	Map<Future<Molecule>, Molecule> drugFut =
215 	    			getFuture2MoleculeMap(chemicalEntity.getDrugs(), ecs);
216         	LOGGER.debug("MOLECULES before submitting inhibitors");
217 		    Map<Future<Molecule>, Molecule> inhFut =
218 		    		getFuture2MoleculeMap(chemicalEntity.getInhibitors(), ecs);
219         	LOGGER.debug("MOLECULES before submitting activators");
220 	        Map<Future<Molecule>, Molecule> actFut =
221 	        		getFuture2MoleculeMap(chemicalEntity.getActivators(), ecs);
222         	LOGGER.debug("MOLECULES before submitting cofactors");
223 	        Map<Future<Molecule>, Molecule> cofFut =
224 	        		getFuture2MoleculeMap(chemicalEntity.getCofactors(), ecs);
225         	LOGGER.debug("MOLECULES before submitting bioactives");
226 	        Map<Future<Molecule>, Molecule> bioactFut =
227 	        		getFuture2MoleculeMap(chemicalEntity.getBioactiveLigands(), ecs);
228 	        // How many jobs have we sent?
229 	        final int numOfMolecules = drugFut.size() + inhFut.size()
230 	        		+ actFut.size() + cofFut.size() + bioactFut.size();
231 	        // Let's retrieve them:
232         	LOGGER.debug("MOLECULES before getting molecules");
233 			for (int i = 0; i < numOfMolecules; i++){
234 	        	Future<Molecule> future = null;
235 	        	try {
236 	        		future = ecs.take();
237 //	        		future = ecs.poll(config.getTimeout(), TimeUnit.MILLISECONDS);
238 	        		if (future != null){
239 	        			final Molecule molecule = future.get();
240 	        			if (molecule != null){
241 		        			// Replace the incomplete molecule with the complete one:
242 	        				if (drugFut.containsKey(future)){
243 	        					drugFut.put(future, molecule);
244 	        				} else if (inhFut.containsKey(future)){
245 	        					inhFut.put(future, molecule);
246 	        				} else if (actFut.containsKey(future)){
247 	        					actFut.put(future, molecule);
248 	        				} else if (cofFut.containsKey(future)){
249 	        					cofFut.put(future, molecule);
250 	        				} else if (bioactFut.containsKey(future)){
251 	        					bioactFut.put(future, molecule);
252 	        				}
253 	        			} else {
254 		    				LOGGER.warn("Molecule retrieved is null!");
255 	        			}
256 	        		} else {
257 	    				LOGGER.warn("Job result not retrieved!");
258 	        		}
259 	        	} catch (Exception e){
260 	            	// Don't stop the others
261 	        		LOGGER.error("Callable " + (i+1) + " of " + numOfMolecules
262 	            			+ " - " + e.getMessage(), e);
263 	        	}
264 	        }
265         	LOGGER.debug("MOLECULES before setting molecules");
266         	// Replace first items in each list with the retrieved ones:
267         	enzymeModel.getMolecule().getDrugs()
268         			.subList(0, drugFut.size()).clear();
269         	enzymeModel.getMolecule().getDrugs().addAll(0, drugFut.values());
270         	enzymeModel.getMolecule().getInhibitors()
271 					.subList(0, inhFut.size()).clear();
272     		enzymeModel.getMolecule().getInhibitors().addAll(0, inhFut.values());
273     		enzymeModel.getMolecule().getActivators()
274     				.subList(0, actFut.size()).clear();
275 			enzymeModel.getMolecule().getActivators().addAll(0, actFut.values());
276 			enzymeModel.getMolecule().getCofactors()
277 					.subList(0, cofFut.size()).clear();
278 			enzymeModel.getMolecule().getCofactors().addAll(0, cofFut.values());
279 			enzymeModel.getMolecule().getBioactiveLigands()
280 					.subList(0, bioactFut.size()).clear();
281 			enzymeModel.getMolecule().getBioactiveLigands().addAll(0, bioactFut.values());
282 	    } finally {
283 	    	pool.shutdown();
284 	    }
285         return enzymeModel;
286     }

288     /**
289      * Prepares callables and sends them to a completion service in order to
290      * retrieve complete entities from ChEBI.
291      * @param molecules the list of molecules to be retrieved from ChEBI.
292      * @param ecs the completion service where the jobs are submitted.
293      * @return a map of <code>Future</code>s to <code>Molecule</code>s
294      */
295     private Map<Future<Molecule>, Molecule> getFuture2MoleculeMap(
296     		List<Molecule> molecules, CompletionService<Molecule> ecs){
297     	Map<Future<Molecule>, Molecule> fut2mol =
298     			new LinkedHashMap<Future<Molecule>, Molecule>();
299         for (Molecule molecule : molecules) {
300         	ChebiWsCallable callable = getCallable(molecule);
301 			if (callable != null){
302 				fut2mol.put(ecs.submit(callable), molecule);
303 			}
304 			// Limit to the first few molecules:
305 			if (fut2mol.size() >= config.getMaxRetrievedMolecules()) break;
306 		}
307         return fut2mol;
308     }

311 	 * Prepares a callable suitable to retrieve a complete Molecule from ChEBI
312 	 * depending on the information available.
313 	 * @param molecule an incomplete Molecule.
314 	 * @return a callable to retrieve the complete Molecule.
315 	 */
316 	private ChebiWsCallable getCallable(Molecule molecule) {
317 		LOGGER.debug("before getting ChEBI callable");
318 		ChebiWsCallable callable = null;
319 		final String drugId = molecule.getId();
320 		if (drugId != null){
321 			callable = drugId.startsWith("CHEBI:")?
322 					new ChebiWsCallable(config, drugId,
323 							SearchCategory.CHEBI_ID):
324 					new ChebiWsCallable(config, drugId,
325 							SearchCategory.DATABASE_LINK_REGISTRY_NUMBER_CITATION);
326 		} else if (molecule.getName() != null){
327 			callable = new ChebiWsCallable(config, molecule.getName(),
328 					SearchCategory.ALL_NAMES);
329 		}
330 		if (callable == null){
331 			LOGGER.warn("The molecule has no id or name");
332 		}
333 		LOGGER.debug("after getting ChEBI callable");
334 		return callable;
335 	}
