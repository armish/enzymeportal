70      protected ChemblBioactivities getBioactivities(String chemblId,
71              SearchBy searchBy)
72      throws ChemblAdapterException {
73          ChemblBioactivities bioactivities = null;
74          InputStream is = null;
75          try {
76              XMLReader xr = XMLReaderFactory.createXMLReader();
77              XPathSAXHandler handler = new XPathSAXHandler(
78                      TARGET_ID_XPATH, COMPOUND_ID_XPATH,
79                      CONFIDENCE_XPATH, ASSAY_TYPE_XPATH);
80              xr.setContentHandler(handler);
81              URL url = new URL(MessageFormat.format(
82                      config.getWsBaseUrl() + BIOACTIVITIES_PATH,
83                      searchBy.name(), chemblId));
84              URLConnection urlCon = url.openConnection();
85              urlCon.setRequestProperty("Accept", "application/xml");
86              is = urlCon.getInputStream();
87              InputSource inputSource = new InputSource(is);
88              xr.parse(inputSource);
89              final Collection<String> targetIds =
90                      handler.getResults().get(TARGET_ID_XPATH);
91              final Collection<String> compoundIds =
92                      handler.getResults().get(COMPOUND_ID_XPATH);
93              final Collection<String> confidences =
94                      handler.getResults().get(CONFIDENCE_XPATH);
95              final Collection<String> assayTypes =
96                      handler.getResults().get(ASSAY_TYPE_XPATH);
97              if (targetIds != null){
98                  bioactivities = new ChemblBioactivities();
99                  Iterator<String> confIter = confidences.iterator();
100                 Iterator<String> asstIter = assayTypes.iterator();
101                 // We get compounds for targets and vice-versa:
102                 switch (searchBy){
103                     case compounds:
104                         for (String targetId : targetIds) {
105                             bioactivities.addBioactivity(targetId,
106                                     Integer.parseInt(confIter.next()),
107                                     asstIter.next());
108                         }
109                         break;
110                     case targets:
111                         Iterator<String> compIter = compoundIds.iterator();
112                         for (String compoundId : compoundIds) {
113                             bioactivities.addBioactivity(compoundId,
114                                     Integer.parseInt(confIter.next()),
115                                     asstIter.next());
116                         }
117                         break;
118                 }
119             }
120         } catch (MalformedURLException e) {
121             throw new ChemblAdapterException(chemblId, e);
122         } catch (IOException e) {
123             throw new ChemblAdapterException(chemblId, e);
124         } catch (SAXException e) {
125             throw new ChemblAdapterException(chemblId, e);
126         } finally {
127             if (is != null) {
128                 try {
129                     is.close();
130                 } catch (IOException e) {
131                     LOGGER.error(e);
132                 }
133             }
134         }
135         return bioactivities;
136     }
