71  	public Collection<SegmentAdapter> getSegments(List<String> segmentIds)
72  	throws MalformedURLException, JAXBException {
73  		// jDAS allows sending the whole list, but the response time is
74  		// proportional to the size of the list, so we use threads:
75  		ExecutorService pool = Executors.newCachedThreadPool();
76  		CompletionService<SegmentAdapter> ecs =
77  				new ExecutorCompletionService<SegmentAdapter>(pool);
78  		Map<Future<SegmentAdapter>, SegmentAdapter> future2segment =
79  				new LinkedHashMap<Future<SegmentAdapter>, SegmentAdapter>();
80  		try {
81  			for (String segmentId : segmentIds) {
82  				Callable<SegmentAdapter> callable = new SimpleDASFeaturesCaller(
83  						IDASFeaturesAdapter.PDBE_DAS_URL, segmentId);
84  				future2segment.put(ecs.submit(callable), null);
85  			}
86  			for (int i = 0; i < segmentIds.size(); i++) {
87  				try {
88  					Future<SegmentAdapter> future = ecs.take(); // TODO: add timeout, from config
89  					if (future != null){
90  						future2segment.put(future, future.get());
91  					} else {
92  						LOGGER.warn("STRUCTURES job result not retrieved!");
93  						future2segment.remove(future);
94  					}
95  				} catch (Exception e){
96  	            	// Don't stop the others
97  	            	LOGGER.error("Callable " + (i+1) + " of " + segmentIds.size()
98  	            			+ " - " + e.getMessage(), e);
99  				}
100 			}
101 			return new ArrayList<SegmentAdapter>(future2segment.values());
102 	    } finally {
103 	    	pool.shutdown();
104 		}
105 	}
