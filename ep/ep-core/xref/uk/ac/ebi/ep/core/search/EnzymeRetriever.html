136     /**
137      * Searches Rhea for the primary UniProt accession in the model and adds the
138      * corresponding reactions if found.
139      * <br><b>WARNING:</b> the added reactions have links only to Reactome and
140      * MACiE. The links are strings containing a complete URL.
141      * @param enzymeModel
142      * @return the same model updated with ReactionPathway objects, one per
143      * 		reaction found.
144      * @throws EnzymeRetrieverException
145      */
146     private EnzymeModel queryRheaWsForReactions(EnzymeModel enzymeModel)
147             throws EnzymeRetrieverException {
148         List<ReactionPathway> reactionPathways = new ArrayList<ReactionPathway>();
149         List<Reaction> reactions;
150         try {
151             reactions = rheaAdapter.getRheasInCmlreact(enzymeModel
152             		.getUniprotaccessions().get(0));
153         } catch (RheaFetchDataException ex) {
154             throw new EnzymeRetrieverException("Query data from Rhea failed! ", ex);
155         }
156         for (Reaction reaction : reactions) {
157             // XXX This adapted reaction will have links only to Reactome and MACiE!:
158             ReactionPathway reactionPathway = rheaAdapter.getReactionPathway(reaction);
159             reactionPathways.add(reactionPathway);
160         }
161         enzymeModel.getReactionpathway().addAll(reactionPathways);
162         return enzymeModel;
163

282     /**
283      * Gets pathways for the enzymatic reactions on the base of xrefs from
284      * Rhea to Reactome.
285      * @param reactionPathways objects containing reactions.
286      * @return the same list of ReactionPathway objects with any added pathways.
287      * @throws EnzymeRetrieverException
288      */
289     protected List<ReactionPathway> getPathwaysFromRheaXref(
290             List<ReactionPathway> reactionPathways) {
291         for (ReactionPathway reactionPathway : reactionPathways) {
292             EnzymeReaction reaction = reactionPathway.getReaction();
293             List<Object> xrefs = reaction.getXrefs();
294             if (xrefs != null) {
295                 for (Object xref : xrefs) {
296                     String x = (String) xref;
297                     // Not all of the xrefs are to Reactome:
298                     final int idStart = x.indexOf("REACT_");
299                     if (idStart == -1) {
300                         continue;
301                     }
302                     // HACK: if xrefs come from Rhea ws as CML, they are complete URLs:
303                     if (x.indexOf("http://") == 0) {
304                         x = x.substring(idStart);
305                     }
306                     // If this fails, at least we have the reactions:
307                     try {
308                         List<Pathway> pathways = getPathwaysByReactomeReactionsId(x);
309                         if (pathways != null) {
310                         	// Check that the retrieved pathway IDs
311                         	// are not there already:
312                         	List<Pathway> newPathways = new ArrayList<Pathway>();
313                         	for (Pathway pathway : pathways) {
314                         		boolean exists = false;
315 								for (Pathway existingPathway : reactionPathway.getPathways()) {
316 									if (existingPathway.getId().equals(pathway.getId())){
317 										exists = true;
318 										break;
319 									}
320 								}
321 								if (!exists) newPathways.add(pathway);
322 							}
323                             reactionPathway.getPathways().addAll(newPathways);
324                         }
325                     } catch (EnzymeRetrieverException e) {
326                         LOGGER.error("Unable to retrieve pathways for " + x, e);
327                     }
328                 }
329             }
330         }
331         return reactionPathways;
332     }

334     private List<ReactionPathway> getReactionPathwaysByUniprotAcc(
335             EnzymeModel enzymeModel)
336             throws EnzymeRetrieverException {
337         //No Rhea reaction, use Reactome reactions
338         String uniprotAccession = enzymeModel.getUniprotaccessions().get(0);
339         List<ReactionPathway> reactionPathwaysFromReactome = null;
340         try {
341             reactionPathwaysFromReactome = biomartAdapter.getReactionsByUniprotAccession(uniprotAccession);
342         } catch (BiomartFetchDataException ex) {
343             throw new EnzymeRetrieverException("Failed to get reactome reactions "
344                     + "from Biomart for uniprot accession " + uniprotAccession, ex);
345         }
346         enzymeModel.setReactionpathway(reactionPathwaysFromReactome);
347         try {
348             getReactomeAdapter().addReactionDescriptions(enzymeModel);
349         } catch (ReactomeServiceException e) {
350             LOGGER.error("Unable to retrieve reaction descriptions", e);
351         }
352         List<ReactionPathway> reactionPathways = enzymeModel.getReactionpathway();
353         getPathwaysFromRheaXref(reactionPathways);
354         return reactionPathways;
355     }

386     public EnzymeModel getMolecules(String uniprotAccession)
387             throws EnzymeRetrieverException {
388         EnzymeModel enzymeModel = null;
389         try {
390             LOGGER.debug("MOLECULES before getting model");
391             enzymeModel = (EnzymeModel) uniprotAdapter.getEnzymeSummaryWithMolecules(uniprotAccession);
392             // The model contains now only drugs, activators and inhibitors.
393             if (megaMapperConnection != null) {
394             	// Get bioactive compounds from ChEMBL:
395                 // Search the mega-map for xrefs from UniProt to ChEMBL:
396                 LOGGER.debug("MOLECULES before getting xrefs to ChEMBL");
397                 //this is the total number of CheMBL molecules found
398                 int totalFound = megaMapperConnection.getMegaMapper().getXrefsSize(
399                         MmDatabase.UniProt, uniprotAccession, MmDatabase.ChEMBL);
400                 List<XRef> chemblXrefs = megaMapperConnection.getMegaMapper().getChMBLXrefs(
401                         MmDatabase.UniProt, uniprotAccession, MmDatabase.ChEMBL);
402                            LOGGER.debug("MOLECULES before getting ChEMBL molecules");
403                 if (chemblXrefs != null) {
404                     Collection<Molecule> chemblDrugs = new ArrayList<Molecule>();
405                     for (XRef chemblXref : chemblXrefs) {
406                         Molecule chemblDrug = new Molecule().withId(chemblXref.getToEntry().getEntryId()).withName(chemblXref.getToEntry().getEntryName());
407                         chemblDrugs.add(chemblDrug);
408                     }
409                     enzymeModel.getMolecule().withBioactiveLigands(chemblDrugs);
410                     enzymeModel.getMolecule().setTotalFound(totalFound);
411                 }
412             }
413             // Get cofactors from IntEnz:
414             enzymeModel.getMolecule().withCofactors(
415             		intenzAdapter.getCofactors(enzymeModel.getEc()));
416             // XXX cofactors are mixed if several EC numbers per UniProt acc
417             // XXX wrong cofactors if OR'ed (ex. EC 1.1.1.1 and P07327)
418             
419             LOGGER.debug("MOLECULES before getting complete entries from ChEBI");
420             getChebiAdapter().getMoleculeCompleteEntries(enzymeModel);
421             LOGGER.debug("MOLECULES before provenance");
422             List<String> prov = new LinkedList<String>();
423             prov.add("ChEBI");
424             prov.add("ChEMBL");
425             // prov.add("RELEASED DATE = " + new Date());
426             prov.add("ChEBI - (Chemical Entities of Biological Interest) is a freely available dictionary of molecular entities focused on ‘small’ chemical compounds.");
427             prov.add("ChEMBL is a database of bioactive drug-like small"
428                     + " molecules, it contains 2-D structures, calculated"
429                     + " properties (e.g. logP, Molecular Weight, Lipinski"
430                     + " Parameters, etc.) and abstracted bioactivities (e.g."
431                     + " binding constants, pharmacology and ADMET data).");
432             if (enzymeModel.getMolecule() != null) {
433                 enzymeModel.getMolecule().setProvenance(prov);
434             }
435         } catch (ChebiFetchDataException ex) {
436             throw new EnzymeRetrieverException(
437                     "Failed to get small molecule details from Chebi", ex);
438         } catch (UniprotWsException e) {
439             throw new EnzymeRetrieverException(
440                     "Unable to get enzyme summary for " + uniprotAccession, e);
441         }
442         return enzymeModel;
443     }
