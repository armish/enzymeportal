13  /**
14   * Processor for EnzymeSummaries which replaces the existing PDBe accessions -
15   * coming from the UniProt web services, including theoretical models - with
16   * those existing in the mega-map - including only experimental ones.
17   * @author rafa
18   *
19   */
20  public class StructuresProcessor implements EnzymeSummaryProcessor {
21  
22  	private MegaMapper mm;
23  	
24  	/**
25  	 * 
26  	 * @param mm the mega-mapper required to get (only experimental) PDB IDs.
27  	 */
28  	public StructuresProcessor(MegaMapper mm){
29  		this.mm = mm;
30  	}
31  	
32  	public void process(EnzymeSummary item) {
33  		String uId = item.getUniprotid();
34  		String uIdPrefix = uId.substring(0, uId.indexOf('_')+1);
35  		Collection<XRef> pdbXrefs = mm.getXrefs(MmDatabase.UniProt, uIdPrefix,
36  				Constraint.STARTS_WITH, MmDatabase.PDB);
37  		List<String> pdbeaccessions = new ArrayList<String>();
38  		if (pdbXrefs != null){
39  			for (XRef xRef : pdbXrefs) {
40  				pdbeaccessions.add(xRef.getToEntry().getEntryId());
41  			}
42  		}
43  		item.setPdbeaccession(pdbeaccessions );
44  	}
