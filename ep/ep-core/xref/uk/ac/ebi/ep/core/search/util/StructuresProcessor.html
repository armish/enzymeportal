14  /**
15   * Processor for EnzymeSummaries which replaces the existing PDBe accessions -
16   * coming from the UniProt web services, including theoretical models - with
17   * those existing in the mega-map - including only experimental ones.
18   * @author rafa
19   * @deprecated processing of summaries after their retrieval is very inefficient
20   */
21  public class StructuresProcessor implements EnzymeSummaryProcessor {
22  
23  	private MegaMapper mm;
24  	
25  	/**
26  	 * 
27  	 * @param mm the mega-mapper required to get (only experimental) PDB IDs.
28  	 */
29  	public StructuresProcessor(MegaMapper mm){
30  		this.mm = mm;
31  	}
32  	
33  	public void process(EnzymeSummary summary) {
34  		String uId = summary.getUniprotid();
35  		String uIdPrefix = uId.substring(0, uId.indexOf('_') + 1);
36  		Collection<XRef> pdbXrefs = mm.getXrefs(MmDatabase.UniProt, uIdPrefix,
37  				Constraint.STARTS_WITH, MmDatabase.PDB);
38  		if (pdbXrefs != null) summary.setPdbeaccession(getToIds(pdbXrefs));
39  
40          for (EnzymeAccession relSp : summary.getRelatedspecies()) {
41              pdbXrefs = mm.getXrefs(MmDatabase.UniProt,
42                      relSp.getUniprotaccessions().get(0), MmDatabase.PDB);
43              if (pdbXrefs != null) relSp.setPdbeaccession(getToIds(pdbXrefs));
44          }
45      }
46  
47      private List<String> getToIds(Collection<XRef> xrefs){
48          List<String> toIds = new ArrayList<String>();
49          for (XRef xRef : xrefs) {
50              toIds.add(xRef.getToEntry().getEntryId());
51          }
52          return toIds;
53      }
