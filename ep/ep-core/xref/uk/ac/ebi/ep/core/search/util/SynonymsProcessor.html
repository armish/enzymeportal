32  	public SynonymsProcessor(List<EnzymeSummary> enzymeSummaries,
33  			IintenzAdapter intenzAdapter) throws MultiThreadingException{
34          Set<String> ecSet = DataTypeConverter.getUniprotEcs(enzymeSummaries);
35          intenzSynonyms = intenzAdapter.getSynonyms(ecSet);
36  	}
37  	
38  	public void process(EnzymeSummary summary) {
39          // Merge UniProt's and IntEnz' synonyms:
40          Set<String> uniqueSyns = new TreeSet<String>(summary.getSynonym());
41          for (String ec : summary.getEc()) {
42              Set<String> ecSynonyms = intenzSynonyms.get(ec);
43              if (ecSynonyms != null) {
44                  uniqueSyns.addAll(ecSynonyms);
45              }
46          }
47          summary.setSynonym(new ArrayList<String>(uniqueSyns));
48  	}
