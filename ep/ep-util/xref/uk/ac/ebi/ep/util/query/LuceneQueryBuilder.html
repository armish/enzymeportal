114     /**
115      * Method to create a Lucene query from the field names and field value.
116      * @param fieldNames The search field names to query.
117      * @param fieldValue The value of the field names.
118      * @return an OR'ed Lucene query
119      */
120     public static String createFieldsQuery(List<String> fieldNames,
121             String fieldValue) {
122         StringBuilder query = new StringBuilder();
123         int listLength = fieldNames.size();
124         int counter = 1;
125         query.append("(");
126         for (String field: fieldNames) {
127             query.append(field);
128             //query.append(":\"");
129             query.append(LUCENE_EQUAL);
130             query.append(LUCENE_QUOTE);
131             query.append(fieldValue);
132             query.append(LUCENE_QUOTE);
133             if (counter <listLength) {
134                 query.append(" OR ");
135             }
136             else {
137                 query.append(")");
138             }
139 
140             counter++;
141         }
142         return query.toString();
143     }

164     public static String addSpeciesFilterQuery(String simpleQuery
165             , String speciesField, Collection<String> speciesList) {
166         StringBuilder sb = new StringBuilder();
167         if (speciesList == null) {
168             return simpleQuery;
169         } else {
170             int listLength = speciesList.size();
171             if (listLength > 0) {
172                 int counter = 1;                
173                 sb.append(simpleQuery);
174                 sb.append(" AND ");
175                 sb.append(speciesField);
176                 sb.append(LUCENE_EQUAL);
177                 sb.append("(");
178                 for (String species: speciesList) {
179                     sb.append(LUCENE_QUOTE);
180                     //sb.append(luceneParser.escapeLuceneSpecialChars(species));
181                     sb.append(species);
182                     sb.append(LUCENE_QUOTE);
183                     if (counter <listLength) {
184                         sb.append(" OR ");
185                     }
186                      counter++;
187                 }
188                 sb.append(")");
189             } else {
190                 return simpleQuery;
191             }
192 
193         }
194         return sb.toString();
195     }

197     public static List<String> escapeSpecies( Collection<String> speciesFilter) {
198         List<String> escapedList = new ArrayList<String>();
199         for (String species:speciesFilter) {
200             String escapedSpecies = luceneParser.escapeLuceneSpecialChars(species);
201             /*Work around to fix the bug in Uniprot API. Any species ended with
202              * " / S288c" has 0 result
203              */
204             String newEscapedSpecies = escapedSpecies.replaceAll(" / S288c", "*/*S288c");
205             escapedList.add(newEscapedSpecies);
206         }
207         return escapedList;
208     }

211      * Create a list of query by id. A query is created for each id. Queries
212      * created by the method can only be used for Uniprot API.
213      * @param idPrefixes
214      * @param speciesFilter
215      * @return

217     public static List<String> createUniprotAPIQueryByIdPrefixes(
218             List<String> idPrefixes, Collection<String> speciesFilter) {
219         List<String> queryList = new ArrayList<String>();
220         List<String> escapedSpeciesFilter = null;
221         if (speciesFilter != null) {
222             escapedSpeciesFilter = escapeSpecies(speciesFilter);
223         }
224         for (String idPrefix : idPrefixes) {
225             StringBuilder sb = new StringBuilder();
226             sb.append(UNIPROT_ID_FIELD);
227             sb.append(LUCENE_EQUAL);
228             sb.append(idPrefix);
229             sb.append(LUCENE_WILDCARD);
230             /*Adding this will make the result list from Uniprot API smaller
231            * than the result list from Ebeye
232            */
233 
234             sb.append(" AND " + ENZYME_FILTER_UNIPROTAPI);
235             queryList.add(
236                     addSpeciesFilterQuery(
237                         sb.toString(), UNIPROT_SPECIES_FIELD, escapedSpeciesFilter));
238         }
239         return queryList;

242     public static String createQueryIN(
243             String fieldName, boolean wildcard, Collection<String> fieldValues) {
244         StringBuilder query = new StringBuilder();
245         query.append(fieldName);
246         query.append(LUCENE_EQUAL);
247         query.append("(");
248         int counter = 1;
249         for (String fieldValue: fieldValues) {
250             query.append(fieldValue);
251             if (wildcard) {
252                 query.append(LUCENE_WILDCARD);
253             }
254             if (counter <fieldValues.size()) {
255                 query.append(" OR ");
256             }
257             counter++;
258         }
259          query.append(")");
260         return query.toString();
261 
262     }

264     /**
265      * Create a list of Lucene queries IN (eg.: id:("PDE7B_HUMAN","PDE7B_MOUSE"))
266      * for a long list of field values. If the list is too long then it is divided
267      * into sub lists. A query is created for each sub list. Every query includes
268      * an "enzyme filter".
269      * @param queryField
270      * @param fieldValues
271      * @param wildcard
272      * @param subListSize
273      * @return
274      */
275     public static List<String> createQueriesIn(
276             String queryField
277             , List<String> fieldValues, boolean wildcard, int subListSize) {
278         List<String> queries = new ArrayList<String>();
279         List<List<String>> subLists = DataTypeConverter
280                 .createSubLists(fieldValues, subListSize);
281         for (List<String> subList: subLists) {
282             StringBuilder sb = new StringBuilder();
283             sb.append(createQueryIN(queryField, wildcard, subList));
284             sb.append( " AND ");
285             sb.append(ENZYME_FILTER);
286             queries.add(sb.toString());
287         }
288         return queries;
289     }
