80          @SuppressWarnings("unchecked")

83          @SuppressWarnings("unchecked")

85                  session.getAttribute(Attribute.basket.name());
86          if (basket == null){
87              basket = Collections.synchronizedMap(
88                      new LinkedHashMap<String, EnzymeSummary>());
89              session.setAttribute(Attribute.basket.name(), basket);
90          }
91          for (String basketId : id.split(";")) {
92              if (checked){
93                  final EnzymeSummary summary = lastSummaries.get(basketId);
94                  if (summary == null){
95                      // build a fresh one:
96                      // TODO
97                  }
98                  basket.put(basketId, summary);
99              } else {
100                 basket.remove(basketId);
101             }
102         }
103         response.setContentType("text/plain");
104         return String.valueOf(basket.size());

113     /**
114      * Compares the two enzymes (accessions) stored in the user session (see
115      * {@link #Attribute.basket.name()}.
116      * @param model the model to populate with the comparison.
117      * @param session the user session containing the enzymes' accessions.
118      * @param accs UniProt accessions of the enzymes to be compared (exactly 2).
119      * @return <code>"comparison"</code> if everything goes well,
120      *      <code>"error"</code> otherwise (less than two enzymes selected to
121      *      compare, for example).
122      * @throws EnzymeRetrieverException in case of problem retrieving the
123      *      enzymes, before the comparison.
124      */
125     @RequestMapping(value="/compare")
126     protected String getComparison(Model model, HttpSession session,
127             @RequestParam(value="acc") String[] accs)
128     throws EnzymeRetrieverException{
129         model.addAttribute("searchModel", newEmptySearchModel());
130         // Filter the incoming accessions, keep only two non-empty:
131         String[] theAccs = new String[2];
132         int j = 0;
133         for (String acc : accs) {
134             if (acc.length() == 0) continue;
135             theAccs[j++] = acc;
136             if (j == 2) break;
137         }
138         ExecutorService pool = null;
139         try {
140             EnzymeModel models[] = new EnzymeModel[2];
141             LOGGER.debug("Getting enzyme models...");
142             pool = Executors.newFixedThreadPool(2);
143             CompletionService<EnzymeModel> cs =
144                     new ExecutorCompletionService<EnzymeModel>(pool);
145             for (String acc : theAccs) {
146                 cs.submit(new EnzymeModelCallable(acc));
147             }
148             for (int i = 0; i < 2; i++) {
149                 EnzymeModel em = cs.take().get();
150                 if (em.getUniprotaccessions().get(0).equals(theAccs[0])){
151                     models[0] = em;
152                 } else {
153                     models[1] = em;
154                 }
155             }
156             LOGGER.debug("Comparison started...");
157             EnzymeComparison comparison =
158                     new EnzymeComparison(models[0], models[1]);
159             LOGGER.debug("Comparison finished");
160             model.addAttribute("comparison", comparison);
161             return "comparison";
162         } catch (Exception e){
163             String errorParam = theAccs[0] + "," + theAccs[1];
164             LOGGER.error("Unable to compare enzymes: " + errorParam, e);
165             model.addAttribute("errorCode", "comparison");
166             model.addAttribute("errorParam", errorParam);
167             return "error";
168         } finally {
169             if (pool != null) pool.shutdownNow();
170         }
171     }

181     private class EnzymeModelCallable implements Callable<EnzymeModel>{
182 
183         private String acc;
184 
185         public EnzymeModelCallable(String acc) {
186             this.acc = acc;
187         }
188 
189         public EnzymeModel call() throws Exception {
190             EnzymeRetriever retriever = null;
191             try {
192                 retriever = new EnzymeRetriever(searchConfig);
193                 retriever.getEbeyeAdapter().setConfig(ebeyeConfig);
194                 retriever.getUniprotAdapter().setConfig(uniprotConfig);
195                 retriever.getIntenzAdapter().setConfig(intenzConfig);
196                 retriever.getReactomeAdapter().setConfig(reactomeConfig);
197                 retriever.getChebiAdapter().setConfig(chebiConfig);
198                 retriever.getBioportalAdapter().setConfig(bioportalConfig);
199                 return retriever.getWholeModel(acc);
200             } finally {
201                 if (retriever != null) retriever.closeResources();
202             }
203         }
