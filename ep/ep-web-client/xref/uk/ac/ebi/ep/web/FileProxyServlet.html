17  /**
18   * Servlet acting as proxy to retrieve files from the file system, out of the
19   * web container.
20   * <br/>
21   * It requires a context parameter <b><code>FileProxyServlet.fs.base</code></b>
22   * set to the path to the directory in the file system used as base for the
23   * file requests. This directory should be readable for the unix user running
24   * the server, and from the machine where it runs.
25   * @author rafa
26   * @since 1.0.25
27   */
28  public class FileProxyServlet extends HttpServlet {
29  
30      private final Logger LOGGER = Logger.getLogger(FileProxyServlet.class);
31  
32      @Override
33      protected void service(HttpServletRequest req, HttpServletResponse resp)
34      throws ServletException, IOException {
35          ApplicationContext context = WebApplicationContextUtils
36                  .getWebApplicationContext(getServletContext());
37          FilesConfig config = context.getBean(FilesConfig.class);
38          File f = new File(config.getBaseDirectory(),
39                  req.getPathInfo().substring(1));
40          if (f.exists() && f.canRead()){
41              String extension = req.getPathInfo()
42                      .substring(req.getPathInfo().lastIndexOf(".") + 1);
43              String contentType = "text/plain;charset=UTF-8";
44              if ("xml".equals(extension)){
45                  contentType = "text/xml;charset=UTF-8";
46              }
47              // Add here any other content types when (if) we serve them.
48              resp.setContentType(contentType);
49              FileInputStream fis = null;
50              BufferedInputStream bis = null;
51              try {
52                  fis = new FileInputStream(f);
53                  bis = new BufferedInputStream(fis);
54                  byte[] buffer = new byte[256];
55                  int r = -1;
56                  while ((r = bis.read(buffer)) != -1){
57                      resp.getOutputStream().write(buffer, 0, r);
58                  }
59                  resp.flushBuffer();
60              } finally {
61                  if (fis != null) fis.close();
62                  if (bis != null) bis.close();
63              }
64          } else {
65              LOGGER.error("File not found: " + f);
66              resp.sendError(HttpServletResponse.SC_NOT_FOUND);
67          }
68      }
