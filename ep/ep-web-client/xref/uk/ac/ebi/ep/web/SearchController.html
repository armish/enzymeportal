127         try {
128             switch (requestedField) {
129                 case proteinStructure:
130                     enzymeModel = retriever.getProteinStructure(accession);
131                     break;
132                 case reactionsPathways:
133                     retriever.getReactomeAdapter().setConfig(reactomeConfig);
134                     enzymeModel = retriever.getReactionsPathways(accession);
135                     break;
136                 case molecules:
137                     retriever.getChebiAdapter().setConfig(chebiConfig);
138                     enzymeModel = retriever.getMolecules(accession);

290     /**
291      * Processes the search request. When user enters a search text and presses
292      * the submit button the request is processed here.
293      *
294      * @param searchModelForm
295      * @param result
296      * @param model
297      * @return
298      */
299     @RequestMapping(value = "/search", method = RequestMethod.POST)
300     public String postSearchResult(SearchModel searchModelForm, BindingResult result,
301             Model model, HttpSession session) {
302         String view = "search";
303         clearHistory(session);
304 
305         if (searchModelForm != null) {
306             try {
307                 SearchParams searchParameters = searchModelForm.getSearchparams();
308                 searchParameters.setSize(searchConfig.getResultsPerPage());
309                 SearchResults resultSet = null;
310                 // See if it is already there, perhaps we are paginating:
311                 @SuppressWarnings("unchecked")
312                 Map<String, SearchResults> prevSearches = (Map<String, SearchResults>) session.getServletContext().getAttribute("searches");
313                 if (prevSearches == null) {
314                     // Map implementation which maintains the order of access:
315                     prevSearches = Collections.synchronizedMap(
316                             new LinkedHashMap<String, SearchResults>(
317                             searchConfig.getSearchCacheSize(), 1, true));
318                     session.getServletContext().setAttribute("searches", prevSearches);
319                 }
320                 resultSet = prevSearches.get(searchParameters.getText().toLowerCase());
321                 if (resultSet == null) {
322                     // Make a new search:
323                     EnzymeFinder finder = new EnzymeFinder(searchConfig);
324                     finder.getEbeyeAdapter().setConfig(ebeyeConfig);
325                     finder.getUniprotAdapter().setConfig(uniprotConfig);
326                     finder.getIntenzAdapter().setConfig(intenzConfig);
327                     try {
328                         resultSet = finder.getEnzymes(searchParameters);
329                         // cache it in the session, making room if necessary:
330                         synchronized (prevSearches) {
331                             while (prevSearches.size() >= searchConfig.getSearchCacheSize()) {
332                                 // remove the eldest:
333                                 prevSearches.remove(prevSearches.keySet().iterator().next());
334                             }
335                             prevSearches.put(searchParameters.getText().toLowerCase(), resultSet);
336                         }
337                     } catch (EnzymeFinderException ex) {
338                         LOGGER.error("Unable to create the result list because an error "
339                                 + "has occurred in the find method! \n", ex);
340                     } finally {
341                         finder.closeResources();
342                     }
343                 }
344 
345                 final int numOfResults = resultSet.getSummaryentries().size();
346                 Pagination pagination = new Pagination(
347                         numOfResults, searchParameters.getSize());
348                 pagination.setFirstResult(searchParameters.getStart());
349 
350                 // Filter:
351                 List<String> speciesFilter = searchParameters.getSpecies();
352                 List<String> compoundsFilter = searchParameters.getCompounds();
353                 List<String> diseasesFilter = searchParameters.getDiseases();
354 
355 //both from auto complete and normal selection. selected items are displayed on top the list and returns back to the orignial list when not selected.
356                 SearchResults searchResults = resultSet;
357                 List<Species> defaultSpeciesList = searchResults.getSearchfilters().getSpecies();
358                 resetSelectedSpecies(defaultSpeciesList);
359 
360                 for (String selectedItems : searchParameters.getSpecies()) {
361 
362                     for (Species theSpecies : defaultSpeciesList) {
363                         if (selectedItems.equals(theSpecies.getScientificname())) {
364                             theSpecies.setSelected(true);
365                         }
366 
367                     }
368                 }
369 
370                 List<Compound> defaultCompoundList = searchResults.getSearchfilters().getCompounds();
371                 resetSelectedCompounds(defaultCompoundList);
372 
373                 for (String SelectedCompounds : searchParameters.getCompounds()) {
374                     for (Compound theCompound : defaultCompoundList) {
375                         if (SelectedCompounds.equals(theCompound.getName())) {
376                             theCompound.setSelected(true);
377                         }
378                     }
379                 }
380 
381                 List<uk.ac.ebi.ep.search.model.Disease> defaultDiseaseList = searchResults.getSearchfilters().getDiseases();
382                 resetSelectedDisease(defaultDiseaseList);
383 
384                 for (String selectedDisease : searchParameters.getDiseases()) {
385                     for (uk.ac.ebi.ep.search.model.Disease disease : defaultDiseaseList) {
386                         if (selectedDisease.equals(disease.getName())) {
387                             disease.setSelected(true);
388                         }
389                     }
390                 }
391 
392 
393 
394                 // list to hold all selected species both from the specie list and auto-complete
395                 Set<String> allSelectedItems = new TreeSet<String>();
396 
397                 //if an item is seleted, then filter the list
398                 if (!speciesFilter.isEmpty() || !compoundsFilter.isEmpty() || !diseasesFilter.isEmpty()) {
399                     List<EnzymeSummary> filteredResults =
400                             new LinkedList<EnzymeSummary>(resultSet.getSummaryentries());
401 
402 
403                     CollectionUtils.filter(filteredResults,
404                             new SpeciesPredicate(speciesFilter));
405                     CollectionUtils.filter(filteredResults,
406                             new CompoundsPredicate(compoundsFilter));
407                     CollectionUtils.filter(filteredResults,
408                             new DiseasesPredicate(diseasesFilter));
409 
410 
411 
412 
413 
414                     allSelectedItems.addAll(compoundsFilter);
415 
416 
417                     allSelectedItems.addAll(diseasesFilter);
418 
419 
420                     allSelectedItems.addAll(speciesFilter);
421 
422 
423                     CollectionUtils.filter(filteredResults, new DefaultPredicate(allSelectedItems));
424 
425 //filtering ends here
426 
427                     // Create a new SearchResults, don't modify the one in session
428                     SearchResults sr = new SearchResults();
429 
430                     // Update the number of results to paginate:
431                     pagination.setNumberOfResults(filteredResults.size());
432 
433                     model.addAttribute("pagination", pagination);
434                     sr.setSearchfilters(resultSet.getSearchfilters());
435                     sr.setSummaryentries(filteredResults);
436                     // show the total number of hits (w/o filtering):
437                     sr.setTotalfound(resultSet.getTotalfound());
438                     searchModelForm.setSearchresults(sr);
439                 } else {
440                     // Show all of them:
441                     searchModelForm.setSearchresults(resultSet);
442                 }
443 
444                 model.addAttribute("searchModel", searchModelForm);
445                 model.addAttribute("pagination", pagination);
446 
447                 addToHistory(session,
448                         "searchparams.text=" + searchParameters.getText());
449             } catch (Throwable e) {
450                 LOGGER.error("Failed search", e);
451                 view = "error";
452             }
453         }
454         return view;
455     }

500     private void addToHistory(HttpSession session, String s) {
501         @SuppressWarnings("unchecked")
502         List<String> history = (List<String>) session.getAttribute("history");
503         if (history == null) {
504             history = new ArrayList<String>();
505             session.setAttribute("history", history);
506         }
507         if (history.isEmpty() || !history.get(history.size() - 1).equals(s)) {
508             history.add(s);
509         }
510     }
