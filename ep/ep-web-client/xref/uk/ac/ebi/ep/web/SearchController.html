96      /**
97       * Process the entry page,
98       *
99       * @param accession The UniProt accession of the enzyme.
100      * @param field the requested tab.
101      * @param model
102      * @return
103      */
104     @RequestMapping(value = "/search/{accession}/{field}")
105     protected String getEnzymeModel(Model model,
106             @PathVariable String accession, @PathVariable String field,
107             HttpSession session) {
108         Field requestedField = Field.valueOf(field);
109         EnzymeRetriever retriever = new EnzymeRetriever(searchConfig);
110         retriever.getEbeyeAdapter().setConfig(ebeyeConfig);
111         retriever.getUniprotAdapter().setConfig(uniprotConfig);
112         retriever.getIntenzAdapter().setConfig(intenzConfig);
113         EnzymeModel enzymeModel = null;
114         String responsePage = ResponsePage.ENTRY.toString();
115         try {
116             switch (requestedField) {
117                 case proteinStructure:
118                     enzymeModel = retriever.getProteinStructure(accession);
119                     break;
120                 case reactionsPathways:
121                     retriever.getReactomeAdapter().setConfig(reactomeConfig);
122                     enzymeModel = retriever.getReactionsPathways(accession);
123                     break;
124                 case molecules:
125                     retriever.getChebiAdapter().setConfig(chebiConfig);
126                     enzymeModel = retriever.getMolecules(accession);
127                     break;
128                 case diseaseDrugs:
129                     retriever.getBioportalAdapter().setConfig(bioportalConfig);
130                     enzymeModel = retriever.getDiseases(accession);
131                     break;
132                 case literature:
133                 	retriever.getLiteratureAdapter().setConfig(literatureConfig);
134                     enzymeModel = retriever.getLiterature(accession);
135                     break;
136                 default:
137                     enzymeModel = retriever.getEnzyme(accession);
138                     requestedField = Field.enzyme;
139                     break;
140             }
141             enzymeModel.setRequestedfield(requestedField.name());
142             model.addAttribute(ENZYME_MODEL, enzymeModel);
143             addToHistory(session, accession);
144         } catch (Exception ex) {
145             // FIXME: this is an odd job to signal an error for the JSP!
146             LOGGER.error("Unable to retrieve the entry!", ex);
147             if (requestedField.getName().equalsIgnoreCase(Field.diseaseDrugs.getName())) {
148                 enzymeModel = new EnzymeModel();
149                 enzymeModel.setName("Diseases");
150                 enzymeModel.setRequestedfield(requestedField.name());
151                 Disease d = new Disease();
152                 d.setName(ERROR);
153                 enzymeModel.getDisease().add(0, d);
154                 model.addAttribute("enzymeModel", enzymeModel);
155                 LOGGER.fatal("Error in retrieving Disease Information");
156             }
157             if (requestedField.getName().equalsIgnoreCase(Field.molecules.getName())) {
158                 enzymeModel = new EnzymeModel();
159                 enzymeModel.setRequestedfield(requestedField.getName());
160                  enzymeModel.setName("Small Molecues");
161                 Molecule molecule = new Molecule();
162                 molecule.setName(ERROR);
163                 ChemicalEntity chemicalEntity = new ChemicalEntity();
164                 chemicalEntity.setDrugs(new CountableMolecules());
165                 chemicalEntity.getDrugs().setMolecule(new ArrayList<Molecule>());
166                 chemicalEntity.getDrugs().getMolecule().add(molecule);
167                 enzymeModel.setMolecule(chemicalEntity);
168                 model.addAttribute(ENZYME_MODEL, enzymeModel);
169                 LOGGER.fatal("Error in retrieving Molecules Information");
170             }
171             if (requestedField.getName().equalsIgnoreCase(Field.enzyme.getName())) {
172 
173                 enzymeModel = new EnzymeModel();
174                 enzymeModel.setRequestedfield(requestedField.getName());
175                  enzymeModel.setName("Enzymes");
176                 Enzyme enzyme = new Enzyme();
177                 enzyme.getEnzymetype().add(0, ERROR);
178                 enzymeModel.setEnzyme(enzyme);
179 
180                 model.addAttribute(ENZYME_MODEL, enzymeModel);
181                 LOGGER.fatal("Error in retrieving Enzymes");
182             }
183             if (requestedField.getName().equalsIgnoreCase(Field.proteinStructure.getName())) {
184                 enzymeModel = new EnzymeModel();
185                 enzymeModel.setRequestedfield(requestedField.getName());
186                  enzymeModel.setName("Protein Structures");
187                 ProteinStructure structure = new ProteinStructure();
188                 structure.setName(ERROR);
189                 enzymeModel.getProteinstructure().add(0, structure);
190 
191                 model.addAttribute(ENZYME_MODEL, enzymeModel);
192                 LOGGER.fatal("Error in retrieving ProteinStructure");
193             }
194             if (requestedField.getName().equalsIgnoreCase(Field.reactionsPathways.getName())) {
195                 enzymeModel = new EnzymeModel();
196 
197                 enzymeModel.setRequestedfield(requestedField.getName());
198                  enzymeModel.setName("Reactions and Pathways");
199                 ReactionPathway pathway = new ReactionPathway();
200                 EnzymeReaction reaction = new EnzymeReaction();
201                 reaction.setName(ERROR);
202                 pathway.setReaction(reaction);
203                 enzymeModel.getReactionpathway().add(0, pathway);
204 
205 
206                 model.addAttribute(ENZYME_MODEL, enzymeModel);
207                 LOGGER.fatal("Error in retrieving Reaction Pathways");
208 
209             }
210             if (requestedField.getName().equalsIgnoreCase(Field.literature.getName())) {
211                 enzymeModel = new EnzymeModel();
212                 enzymeModel.setRequestedfield(requestedField.getName());
213                  enzymeModel.setName("Literatures");
214 
215                 enzymeModel.getLiterature().add(0, ERROR);
216 
217                 model.addAttribute(ENZYME_MODEL, enzymeModel);
218                 LOGGER.fatal("Error in retrieving Literature Information");
219 
220             }
221 
222         } finally {
223             retriever.closeResources();
224         }
225         return responsePage;
226     }

346     @RequestMapping(value = "/search", method = RequestMethod.POST)
347     public String postSearchResult(SearchModel searchModel, Model model,
348     		HttpSession session) {
349         String view = "error";
350         try {
351 	        // See if it is already there, perhaps we are paginating:
352 	        Map<String, SearchResults> prevSearches =
353 	        		getPreviousSearches(session.getServletContext());
354                                 
355 	        String searchKey = getSearchKey(searchModel.getSearchparams());
356                 
357                
358 	        SearchResults results = prevSearches.get(searchKey);
359 	        if (results == null){
360 	        	// New search:
361 	            clearHistory(session);
362                  
363 	            switch (searchModel.getSearchparams().getType()) {
364                          
365 	    		case KEYWORD:
366 	            	results = searchKeyword(searchModel.getSearchparams());
367 	                cacheSearch(session.getServletContext(), searchKey, results);
368 	                addToHistory(session, "searchparams.text=" + searchKey);
369 	    			break;
370 	    		case SEQUENCE:
371 	            	view = searchSequence(model, searchModel);
372                         // cacheSearch(session.getServletContext(), searchKey, results);
373 	                addToHistory(session, "searchparams.sequence=" + searchKey);
374 	    			break;
375 	    		case COMPOUND:
376 	//            	view = postCompoundSearch(model, searchModel);
377 	    			break;
378                             default:
379 	    		}
380 	         }
381                
382 	        if (results != null) { // something to show
383 	        	searchModel.setSearchresults(results);
384 	        	applyFilters(searchModel);
385 	        	model.addAttribute("searchModel", searchModel);
386 	        	model.addAttribute("pagination", getPagination(searchModel));
387                         clearHistory(session);
388                         if(searchModel.getSearchparams().getType().equals(SearchParams.SearchType.KEYWORD)){
389                            addToHistory(session, "searchparams.text=" + searchKey); 
390                         }
391                          if(searchModel.getSearchparams().getType().equals(SearchParams.SearchType.SEQUENCE)){
392                            addToHistory(session, "searchparams.sequence=" + searchKey); 
393                         }
394                         //addToHistory(session, "searchparams.text=" + searchKey);
395 	        	view = "search";
396 	        }
397         } catch (Throwable t){
398             LOGGER.error("one of the search params (Text or Sequence is :", t);
399         }

405      * @param searchParameters the search parameters.

407      */
408     private SearchResults searchKeyword(SearchParams searchParameters){
409     	SearchResults results = null;
410         EnzymeFinder finder = new EnzymeFinder(searchConfig);
411         finder.getEbeyeAdapter().setConfig(ebeyeConfig);
412         finder.getUniprotAdapter().setConfig(uniprotConfig);
413         finder.getIntenzAdapter().setConfig(intenzConfig);

419         } finally {
420             finder.closeResources();
421         }
422     	return results;

592     /**
593      * Retrieves any previous searches stored in the application context.
594      * @param servletContext the application context.
595      * @return a map of searches to results.
596      */
597     @SuppressWarnings("unchecked")
598 	private Map<String, SearchResults> getPreviousSearches(
599 			ServletContext servletContext){
600         Map<String, SearchResults> prevSearches = (Map<String, SearchResults>)
601         		servletContext.getAttribute("searches");
602         if (prevSearches == null) {
603             // Map implementation which maintains the order of access:
604             prevSearches = Collections.synchronizedMap(
605                     new LinkedHashMap<String, SearchResults>(
606                     searchConfig.getSearchCacheSize(), 1, true));
607             servletContext.setAttribute("searches", prevSearches);
608         }
609         return prevSearches;
610     }

612     /**
613      * Stores a search result in the application context.
614      * @param servletContext the application context.
615      * @param searchKey the key to use for the search results in the table.
616      * @param searchResult the search results.
617      */
618     private void cacheSearch(ServletContext servletContext, String searchKey,
619     		SearchResults searchResult){
620         Map<String, SearchResults> prevSearches =
621         		getPreviousSearches(servletContext);
622     	synchronized (prevSearches) {
623             while (prevSearches.size() >= searchConfig.getSearchCacheSize()) {
624                 // remove the eldest:
625                 prevSearches.remove(prevSearches.keySet().iterator().next());
626             }
627             prevSearches.put(searchKey, searchResult);
628     	}
629     }

660     /**
661      * Sends a search to a BLAST server. The job ID returned is stored in the
662      * model as <code>jobId</code> attribute.
663      * @param model the Spring model.
664      * @param searchModel the EP search model.
665      * @return the view according to the search: <code>running</code> if
666      * 		everything went ok and the search job is running, <code>error</code>
667      * 		otherwise.
668      */
669     private String searchSequence(Model model, SearchModel searchModel){
670     	String view = "error";
671     	try {
672         	EnzymeFinder finder = new EnzymeFinder(searchConfig);
673 			String sequence = searchModel.getSearchparams().getSequence()
674 					.trim().toUpperCase();
675 			String jobId = finder.blast(sequence);
676 			searchModel.getSearchparams().setPrevioustext(sequence);
677 			model.addAttribute("jobId", jobId);
