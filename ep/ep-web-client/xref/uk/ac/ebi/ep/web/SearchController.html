110     /**
111      * Process the entry page,
112      *
113      * @param accession The UniProt accession of the enzyme.
114      * @param field the requested tab.
115      * @param model
116      * @return
117      */
118     @RequestMapping(value = "/search/{accession}/{field}")
119     protected String getEnzymeModel(Model model,
120             @PathVariable String accession, @PathVariable String field,
121             HttpSession session) {
122         Field requestedField = Field.valueOf(field);
123         EnzymeRetriever retriever = new EnzymeRetriever(searchConfig);
124         retriever.getEbeyeAdapter().setConfig(ebeyeConfig);
125         retriever.getUniprotAdapter().setConfig(uniprotConfig);
126         retriever.getIntenzAdapter().setConfig(intenzConfig);
127         EnzymeModel enzymeModel = null;
128         String responsePage = ResponsePage.ENTRY.toString();
129         try {
130             switch (requestedField) {
131                 case proteinStructure:
132                     enzymeModel = retriever.getProteinStructure(accession);
133                     break;
134                 case reactionsPathways:
135                     retriever.getReactomeAdapter().setConfig(reactomeConfig);
136                     enzymeModel = retriever.getReactionsPathways(accession);
137                     break;
138                 case molecules:
139                     retriever.getChebiAdapter().setConfig(chebiConfig);
140                     enzymeModel = retriever.getMolecules(accession);
141                     break;
142                 case diseaseDrugs:
143                     enzymeModel = retriever.getDiseases(accession);
144                     break;
145                 case literature:
146                 	retriever.getLiteratureAdapter().setConfig(literatureConfig);
147                     enzymeModel = retriever.getLiterature(accession);
148                     break;
149                 default:
150                     enzymeModel = retriever.getEnzyme(accession);
151                     requestedField = Field.enzyme;
152                     break;
153             }
154             enzymeModel.setRequestedfield(requestedField.name());
155             model.addAttribute(ENZYME_MODEL, enzymeModel);
156             addToHistory(session, accession);
157         } catch (Exception ex) {
158             LOGGER.error("Unable to retrieve the entry!", ex);
159             if (requestedField.getName().equalsIgnoreCase(Field.diseaseDrugs.getName())) {
160                 enzymeModel = new EnzymeModel();
161                 enzymeModel.setRequestedfield(requestedField.name());
162                 Disease d = new Disease();
163                 d.setName(ERROR);
164                 enzymeModel.getDisease().add(0, d);
165                 model.addAttribute("enzymeModel", enzymeModel);
166                 LOGGER.fatal("Error in retrieving Disease Information");
167             }
168             if (requestedField.getName().equalsIgnoreCase(Field.molecules.getName())) {
169                 enzymeModel = new EnzymeModel();
170                 enzymeModel.setRequestedfield(requestedField.getName());
171                 Molecule molecule = new Molecule();
172                 molecule.setName(ERROR);
173                 ChemicalEntity chemicalEntity = new ChemicalEntity();
174                 chemicalEntity.getDrugs().add(0, molecule);
175                 enzymeModel.setMolecule(chemicalEntity);
176                 model.addAttribute(ENZYME_MODEL, enzymeModel);
177                 LOGGER.fatal("Error in retrieving Molecules Information");
178             }
179             if (requestedField.getName().equalsIgnoreCase(Field.enzyme.getName())) {
180 
181                 enzymeModel = new EnzymeModel();
182                 enzymeModel.setRequestedfield(requestedField.getName());
183                 Enzyme enzyme = new Enzyme();
184                 enzyme.getEnzymetype().add(0, ERROR);
185                 enzymeModel.setEnzyme(enzyme);
186 
187                 model.addAttribute(ENZYME_MODEL, enzymeModel);
188                 LOGGER.fatal("Error in retrieving Enzymes");
189             }
190             if (requestedField.getName().equalsIgnoreCase(Field.proteinStructure.getName())) {
191                 enzymeModel = new EnzymeModel();
192                 enzymeModel.setRequestedfield(requestedField.getName());
193                 ProteinStructure structure = new ProteinStructure();
194                 structure.setName(ERROR);
195                 enzymeModel.getProteinstructure().add(0, structure);
196 
197                 model.addAttribute(ENZYME_MODEL, enzymeModel);
198                 LOGGER.fatal("Error in retrieving ProteinStructure");
199             }
200             if (requestedField.getName().equalsIgnoreCase(Field.reactionsPathways.getName())) {
201                 enzymeModel = new EnzymeModel();
202 
203                 enzymeModel.setRequestedfield(requestedField.getName());
204                 ReactionPathway pathway = new ReactionPathway();
205                 EnzymeReaction reaction = new EnzymeReaction();
206                 reaction.setName(ERROR);
207                 pathway.setReaction(reaction);
208                 enzymeModel.getReactionpathway().add(0, pathway);
209 
210 
211                 model.addAttribute(ENZYME_MODEL, enzymeModel);
212                 LOGGER.fatal("Error in retrieving Reaction Pathways");
213 
214             }
215             if (requestedField.getName().equalsIgnoreCase(Field.literature.getName())) {
216                 enzymeModel = new EnzymeModel();
217                 enzymeModel.setRequestedfield(requestedField.getName());
218 
219                 enzymeModel.getLiterature().add(0, ERROR);
220 
221                 model.addAttribute(ENZYME_MODEL, enzymeModel);
222                 LOGGER.fatal("Error in retrieving Literature Information");
223 
224             }
225 
226         } finally {
227             retriever.closeResources();
228         }
229         return responsePage;
230     }

413      * Searches by keyword.
414      * @param searchParameters the search parameteres.
415      * @return the search results.
416      */
417     private SearchResults searchKeyword(SearchParams searchParameters){
418     	SearchResults results = null;
419         EnzymeFinder finder = new EnzymeFinder(searchConfig);
420         finder.getEbeyeAdapter().setConfig(ebeyeConfig);
421         finder.getUniprotAdapter().setConfig(uniprotConfig);
422         finder.getIntenzAdapter().setConfig(intenzConfig);
423         try {
424             results = finder.getEnzymes(searchParameters);
425         } catch (EnzymeFinderException ex) {
426             LOGGER.error("Unable to create the result list because an error "
427                     + "has occurred in the find method! \n", ex);
428         } finally {
429             finder.closeResources();
430         }
431     	return results;
432     }

457     /**
458      * Applies filters taken from the search parameters to the search results.
459      * @param searchModel
460      */
461 
462     private void applyFilters(SearchModel searchModel){
463 
464                 
465                        if (searchModel != null) {
466         
467                 SearchParams searchParameters = searchModel.getSearchparams();
468                 searchParameters.setSize(searchConfig.getResultsPerPage());
469                 SearchResults resultSet = searchModel.getSearchresults();
470 
471 
472                 final int numOfResults = resultSet.getSummaryentries().size();
473                 Pagination pagination = new Pagination(
474                         numOfResults, searchParameters.getSize());
475                 pagination.setFirstResult(searchParameters.getStart());
476 
477                 // Filter:
478                 List<String> speciesFilter = searchParameters.getSpecies();
479                 List<String> compoundsFilter = searchParameters.getCompounds();
480                 List<String> diseasesFilter = searchParameters.getDiseases();
481 
482 //both from auto complete and normal selection. selected items are displayed on top the list and returns back to the orignial list when not selected.
483                 SearchResults searchResults = resultSet;
484                 List<Species> defaultSpeciesList = searchResults.getSearchfilters().getSpecies();
485                 resetSelectedSpecies(defaultSpeciesList);
486 
487                 for (String selectedItems : searchParameters.getSpecies()) {
488 
489                     for (Species theSpecies : defaultSpeciesList) {
490                         if (selectedItems.equals(theSpecies.getScientificname())) {
491                             theSpecies.setSelected(true);
492                         }
493 
494                     }
495                 }
496 
497                 List<Compound> defaultCompoundList = searchResults.getSearchfilters().getCompounds();
498                 resetSelectedCompounds(defaultCompoundList);
499 
500                 for (String SelectedCompounds : searchParameters.getCompounds()) {
501                     for (Compound theCompound : defaultCompoundList) {
502                         if (SelectedCompounds.equals(theCompound.getName())) {
503                             theCompound.setSelected(true);
504                         }
505                     }
506                 }
507 
508                 List<uk.ac.ebi.ep.search.model.Disease> defaultDiseaseList = searchResults.getSearchfilters().getDiseases();
509                 resetSelectedDisease(defaultDiseaseList);
510 
511                 for (String selectedDisease : searchParameters.getDiseases()) {
512                     for (uk.ac.ebi.ep.search.model.Disease disease : defaultDiseaseList) {
513                         if (selectedDisease.equals(disease.getName())) {
514                             disease.setSelected(true);
515                         }
516                     }
517                 }
518 
519                 // list to hold all selected species both from the specie list and auto-complete
520                 Set<String> allSelectedItems = new TreeSet<String>();
521 
522                 //if an item is seleted, then filter the list
523                 if (!speciesFilter.isEmpty() || !compoundsFilter.isEmpty() || !diseasesFilter.isEmpty()) {
524                     List<EnzymeSummary> filteredResults =
525                             new LinkedList<EnzymeSummary>(resultSet.getSummaryentries());
526 
527 
528                     CollectionUtils.filter(filteredResults,
529                             new SpeciesPredicate(speciesFilter));
530                     CollectionUtils.filter(filteredResults,
531                             new CompoundsPredicate(compoundsFilter));
532                     CollectionUtils.filter(filteredResults,
533                             new DiseasesPredicate(diseasesFilter));
534 
535 
536 
537 
538 
539                     allSelectedItems.addAll(compoundsFilter);
540 
541 
542                     allSelectedItems.addAll(diseasesFilter);
543 
544 
545                     allSelectedItems.addAll(speciesFilter);
546 
547 
548                     CollectionUtils.filter(filteredResults, new DefaultPredicate(allSelectedItems));
549                     
550                     
551                     //adapting the sequece code
552                     
553                        // Create a new SearchResults, don't modify the one in session
554                     SearchResults sr = new SearchResults();
555 
556                     // Update the number of results to paginate:
557                     pagination.setNumberOfResults(filteredResults.size());
558 
559                     //model.addAttribute("pagination", pagination);
560                     sr.setSearchfilters(resultSet.getSearchfilters());
561                     sr.setSummaryentries(filteredResults);
562                     // show the total number of hits (w/o filtering):
563                     sr.setTotalfound(resultSet.getTotalfound());
564                     searchModel.setSearchresults(sr);
565 
566 //filtering ends here
567 
568         }
569                 
570 
571                 }
572 
573     }

575     /**
576      * Retrieves any previous searches stored in the application context.
577      * @param servletContext the application context.
578      * @return a map of searches to results.
579      */
580     @SuppressWarnings("unchecked")
581 	private Map<String, SearchResults> getPreviousSearches(
582 			ServletContext servletContext){
583         Map<String, SearchResults> prevSearches = (Map<String, SearchResults>)
584         		servletContext.getAttribute("searches");
585         if (prevSearches == null) {
586             // Map implementation which maintains the order of access:
587             prevSearches = Collections.synchronizedMap(
588                     new LinkedHashMap<String, SearchResults>(
589                     searchConfig.getSearchCacheSize(), 1, true));
590             servletContext.setAttribute("searches", prevSearches);
591         }
592         return prevSearches;
593     }

595     /**
596      * Stores a search result in the application context.
597      * @param servletContext the application context.
598      * @param searchKey the key to use for the search results in the table.
599      * @param searchResult the search results.
600      */
601     private void cacheSearch(ServletContext servletContext, String searchKey,
602     		SearchResults searchResult){
603         Map<String, SearchResults> prevSearches =
604         		getPreviousSearches(servletContext);
605     	synchronized (prevSearches) {
606             while (prevSearches.size() >= searchConfig.getSearchCacheSize()) {
607                 // remove the eldest:
608                 prevSearches.remove(prevSearches.keySet().iterator().next());
609             }
610             prevSearches.put(searchKey, searchResult);
611     	}
612     }

643     /**
644      * Sends a search to a BLAST server. The job ID returned is stored in the
645      * model as <code>jobId</code> attribute.
646      * @param model the Spring model.
647      * @param searchModel the EP search model.
648      * @return the view according to the search: <code>running</code> if
649      * 		everything went ok and the search job is running, <code>error</code>
650      * 		otherwise.
651      */
652     private String searchSequence(Model model, SearchModel searchModel){
653     	String view = "error";
654     	try {
655         	EnzymeFinder finder = new EnzymeFinder(searchConfig);
656 			String sequence = searchModel.getSearchparams().getSequence()
657 					.trim().toUpperCase();
658 			String jobId = finder.blast(sequence);
659 			searchModel.getSearchparams().setPrevioustext(sequence);
660 			model.addAttribute("jobId", jobId);
661 			LOGGER.debug("BLAST job running: " + jobId);
662 			view = "running";
663 		} catch (NcbiBlastClientException e) {
664 			LOGGER.error(e);
665 		}
666     	return view;
667     }
