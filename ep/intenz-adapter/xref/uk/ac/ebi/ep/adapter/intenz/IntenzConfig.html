64      /**
65       * Reads the configuration from a file.
66       * @param filePath the path to a file with the configuration properties.
67       *      Searched first in the classpath, then if not found in the current
68       *      working directory.
69       * @return a configuration for the IntEnz proxy with the properties values

71       * @throws IOException in case of problems reading from file.
72       * @since 1.0.6
73       * @see #readFromInputStream(InputStream) Properties' keys.
74       */
75      public static IntenzConfig readFromFile(String filePath) throws IOException{
76          InputStream is = null;
77          try {
78              is = IntenzConfig.class.getClassLoader().getResourceAsStream(filePath);
79              if (is == null){
80                  LOGGER.info(filePath + " not found in the classpath,"
81                          + " trying the working directory...");
82                  is = new FileInputStream(filePath);
83              }
84              return readFromInputStream(is);
85          } finally {
86              if (is != null) try {
87                  is.close();
88              } catch (IOException e) {
89                  LOGGER.error("Unable to close input stream", e);
90              }
91          }
92      }

94      /**
95       * Reads the configuration from an input stream. The properties' keys are:
96       * <ul>
97       *  <li><code>ep.intenz.ws.timeout</code>: (integer) maximum time
98       *      (ms) to get a response from the web service.</li>
99       *  <li><code>ep.intenz.ws.xml.url</code>: (string) URL template
100      *      to retrieve IntEnz entries as IntEnzXML from the web service. It
101      *      should have the placeholders {0}, {1}, {2} and {3} for ec1, ec2, ec3
102      *      and ec4 respectively.</li>
103      *  <li><code>ep.intenz.ec.url</code>: (string) base URL for EC
104      *      numbers in the IntEnz website (the complete URL is built just adding
105      *      the EC number at the end).</li>
106      * </ul>
107      * @param is an input stream.
108      * @return a configuration for the IntEnz proxy with the properties values
109      *      from the file, or the default ones for those missing.
110      * @throws IOException in case of problem reading the properties from file.
111      * @since 1.0.6
112      */
113     protected static IntenzConfig readFromInputStream(InputStream is)
114     throws IOException{
115         Properties props = new Properties();
116         props.load(is);
117         IntenzConfig config = new IntenzConfig();
118         try {
119             config.setTimeout(Integer.valueOf(props.getProperty(TIMEOUT)));
120         } catch (NumberFormatException e){
121             LOGGER.error("Using default for invalid value " + TIMEOUT
122                     + "=" + props.getProperty(TIMEOUT));
123         }
124         config.setIntenzXmlUrl(props.getProperty(XML_URL,
125                 config.getIntenzXmlUrl()));
126         config.setEcBaseUrl(props.getProperty(EC_URL, config.getEcBaseUrl()));
127         return config;
128     }
