43      /**
44       * Searches CiteXplore for citations.
45       * @return a collection of citations from CiteXplore web services, or
46       *      <code>null</code> if none found.
47       * @throws Exception
48       * @since 1.0.7 (refactored out of {@link #call()})
49       */
50      Collection<Result> searchCitations() throws Exception {
51          CitexploreWSClient citexploreClient = null;
52          try {
53              citexploreClient =
54                      CitexploreWSClientPool.getInstance().borrowObject();
55              citexploreClient.setConnectTimeout(connectTimeout);
56              citexploreClient.setReadTimeout(readTimeout);
57              citexploreClient.setMaxCitations(maxCitations);
58              return citexploreClient.searchCitations(query);
59          } finally {
60              if (citexploreClient != null){
61                  CitexploreWSClientPool.getInstance()
62                          .returnObject(citexploreClient);
63              }
64          }
65      }
