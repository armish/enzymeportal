68  	/**
69  	 * Extracts literature information from the segments returned by DAS.
70  	 * @param segments
71  	 * @return
72  	 */
73  	protected Set<Result> getLiterature(List<SegmentAdapter> segments) {
74  		Set<Result> citations = new HashSet<Result>();
75  		Map<String, FeatureAdapter> authorsFeatures =
76  		        new HashMap<String, FeatureAdapter>();
77          Map<String, FeatureAdapter> primaryCitationFeatures =
78                  new HashMap<String, FeatureAdapter>();
79          Map<String, String> pubMedIds = new HashMap<String, String>();
80  		for (SegmentAdapter segment : segments) {
81  			try {
82  				for (FeatureAdapter feature : segment.getFeature()) {
83  				    if (isDepositedBy(feature)){
84                          String pdbId = feature.getId().replace("-authors", "");
85                          authorsFeatures.put(pdbId, feature);
86                      } else if (isPrimaryCitation(feature)){
87                          String pdbId = feature.getId().replace("-citation", "");
88                          primaryCitationFeatures.put(pdbId, feature);
89                          String pubMedId = null;
90  						if (feature.getLinks().size() == 0){
91  							LOGGER.warn(feature.getId()
92  									+ ": link (PMID) not available");
93  						} else {
94                              pubMedId = getPubMedId(
95                                      feature.getLinks().get(0).getHref());
96                          }
97                          pubMedIds.put(pdbId, pubMedId);
98  					}
99  				}
100 				for (String pdbId : primaryCitationFeatures.keySet()){
101                     final String pubMedId = pubMedIds.get(pdbId);
102                     if (pubMedId == null) continue;
103                     Result citation = buildCitation(
104                             primaryCitationFeatures.get(pdbId),
105                             authorsFeatures.get(pdbId),
106                             pubMedId);
107                     citations.add(citation);
108                 }
109 			} catch (ValidationException e) {
110 				LOGGER.error("Unable to get citations from DAS for "
111 						+ segment.getId(), e);
112 			}
113 		}
114 		return citations;
115 	}

117     private boolean isPrimaryCitation(FeatureAdapter feature) {

119                 && feature.getLabel().equals("Primary Citation");
120     }

124                 && feature.getLabel().equals("Deposited by");

127     private Result buildCitation(FeatureAdapter citationsFeature,
128             FeatureAdapter authorsFeature, String pubMedId) {
129 		Result citation = new Result();
130 		JournalInfo issue = new JournalInfo();
131 		Journal journal = new Journal();
132 		issue.setJournal(journal);
133 		citation.setJournalInfo(issue);
134 		citation.setTitle(citationsFeature.getNotes().get(0));
135 		parseJournal(citationsFeature.getNotes().get(1), citation);
136 		citation.setAuthorList(new AuthorsList());
137 		citation.getAuthorList().getAuthor().addAll(
138                 parseAuthors(authorsFeature.getNotes().get(0)));
139 		citation.setSource(DataSource.MED.name());
140 		citation.setId(pubMedId);
141 		return citation;
142 	}

144 	protected Result getCitationFromCitexplore(String pubMedId) {
145 		Result citation = null;
146 		CitexploreWSClient citexploreClient = null;
147 		try {
148 			citexploreClient =
149 					CitexploreWSClientPool.getInstance().borrowObject();
150 			citexploreClient.setConnectTimeout(connectTimeout);
151 			citexploreClient.setReadTimeout(readTimeout);
152 			citation = citexploreClient.retrieveCitation(DataSource.MED,
153 			        pubMedId);
154 		} catch (Exception e) {
155 			LOGGER.error("Unable to get CiteXplore citation for " + pubMedId,
156 			        e);
157 		} finally {
158 			if (citexploreClient != null){
159 				try {
160 					CitexploreWSClientPool.getInstance().returnObject(citexploreClient);
161 				} catch (Exception e) {
162 					LOGGER.error("Unable to return CiteXplore client", e);
163 				}
164 			}
165 		}
166 		return citation;
167 	}

170 	 * Extracts the PubMed ID from the URL returned by DAS.
171 	 * @param url
172 	 * @return a PubMed ID
173 	 */
174 	protected String getPubMedId(String url) {
175 		final String pubMedParam = "Citation_pubmed_id=";
176 		int index = url.lastIndexOf(pubMedParam);
177 		return index > -1?
178 				url.substring(index + pubMedParam.length()):
179 				null;
180 	}

182 	protected List<Authors> parseAuthors(String authors){
183 	    List<Authors> l = new ArrayList<Authors>();
184 		for (String sAuthor : authors.split("(?<=\\.), ")) {
185             Authors author = new Authors();
186             author.setFullName(sAuthor);
187 			l.add(author);
188 		}
189 		return l;
190 	}

192 	/**
193 	 * Populates a Result with journal data from a string.
194 	 * @param journal String with the standard notation of the publication,
195 	 * 		which comes from DAS as
196 	 * 		<code>[journal] vol:[volume] page:[pages] ([year])</code>
197 	 * @param citation A CiteXplore citation.
198 	 */
199 	protected void parseJournal(String journal, Result citation) {
200 		Pattern p = Pattern.compile("(.+?)(?: vol:(\\d+))?(?: page:(\\d+-\\d+))? \\((\\d{4})\\)");
201 		Matcher m = p.matcher(journal);
202 		if (m.matches()){
203 			citation.getJournalInfo().getJournal().setTitle(m.group(1));
204 			citation.getJournalInfo().setVolume(m.group(2));
205 			citation.setPageInfo(m.group(3));
206             final String pubYear = m.group(4);
207             citation.getJournalInfo().setYearOfPublication(
208                     Short.valueOf(pubYear));
209 			citation.setPubYear(pubYear);
210 		} else {
211 			LOGGER.error("Format not recognised for DAS citation: " + journal);
212 		}
213 	}
