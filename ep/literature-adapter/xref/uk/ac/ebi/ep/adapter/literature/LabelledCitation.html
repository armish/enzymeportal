46      /**
47       * Equals method based on bibliography identifier, namely PubMed ID.
48       * If missing from both compared objects, the title is used.<br/>
49       * This method does not take labels into account.
50       */
51      @Override
52      public boolean equals(Object o) {
53          if (!(o instanceof LabelledCitation)) return false;
54          if (o == this) return true;
55          LabelledCitation other = (LabelledCitation) o;
56          String tcei = this.citation.getId();
57          String ocei = other.citation.getId();
58          if (tcei == null ^ ocei == null) return false;
59          if (tcei != null && ocei != null) {
60              String tcds = this.citation.getSource();
61              String ocds = other.citation.getSource();
62              return tcei.equals(ocei) && tcds.equals(ocds);
63          }
64          return this.citation.getTitle().equals(other.citation.getTitle());
65      }

67      @Override
68      public int hashCode() {
69          int hash = 17;
70          hash = hash * 17;
71          if (citation.getSource() != null) {
72              hash += citation.getSource().hashCode();
73          }
74          hash = hash * 17;
75          if (citation.getId() != null) {
76              hash += citation.getId().hashCode();
77          }
78          hash = hash * 17 + citation.getTitle().hashCode();
79          return hash;
80      }

82      /**
83       * By default, citations are sorted by date of creation (if not available,
84       * year of publication).
85       */
86      public int compareTo(LabelledCitation other) {
87          if (other == this) return 0;
88          if (this.citation.getDateOfCreation() == null){
89              if (other.citation.getDateOfCreation() == null){
90                  // Fall back to publication year:
91                  if (this.citation.getPubYear() == null){
92                      if (other.citation.getPubYear() == null) return 0;
93                      else return 1;
94                  } else if (other.citation.getPubYear() == null){
95                      return -1;
96                  }
97                  return Short.valueOf(this.citation.getPubYear())
98                      .compareTo(Short.valueOf(other.citation.getPubYear()));
99              } else return 1;
100         } else if (other.citation.getDateOfCreation() == null){
101             return -1;
102         }
103         return this.citation.getDateOfCreation()
104                 .compare(other.citation.getDateOfCreation());
105     }
