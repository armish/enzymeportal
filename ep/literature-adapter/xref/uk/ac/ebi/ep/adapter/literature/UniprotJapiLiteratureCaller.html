63  	public Set<uk.ac.ebi.cdb.webservice.Result> call()
64  	throws Exception {
65  		// CiteXplore citations:
66  		HashSet<uk.ac.ebi.cdb.webservice.Result> cxCits = null;
67  		UniProtQueryService uniProtQueryService =
68  				UniProtJAPI.factory.getUniProtQueryService();
69  		Query query = UniProtQueryBuilder.buildIDListQuery(
70  				Arrays.asList(new String[]{ uniprotId }));
71  		AttributeIterator<UniProtEntry> it =
72  				uniProtQueryService.getAttributes(query, "ognl:citations");
73  		if (it.hasNext()){
74  			cxCits = new HashSet<uk.ac.ebi.cdb.webservice.Result>();
75  			for (Attribute att: it){
76  				// UniProt citations:
77  				@SuppressWarnings("unchecked")
78  				List<Citation> upCits = (List<Citation>) att.getValue();
79  				for (Citation upCit :upCits){
80  					uk.ac.ebi.cdb.webservice.Result cxCit = null;
81  					// Try to get it from CiteXplore:
82  					cxCit = getCitationFromCitexplore(upCit);
83  					if (cxCit == null){
84  						// Otherwise, build one:
85  						LOGGER.warn("No literature identifier found for one citation: " + uniprotId);
86  						//cxCit = buildCitation(upCit);
87  					} else {
88  						cxCits.add(cxCit);
89  					}
90  				}
91  			}
92  		} else {
93  			LOGGER.warn("No citations retrieved from " + uniprotId);
94  		}
95  		return cxCits;
96  	}

98  	private uk.ac.ebi.cdb.webservice.Result buildCitation(Citation upCit) {
99  		uk.ac.ebi.cdb.webservice.Result cxCit =
100 				new uk.ac.ebi.cdb.webservice.Result();
101 		JournalInfo issue = new JournalInfo();
102 		Journal journal = new Journal();
103 		issue.setJournal(journal);
104 		cxCit.setJournalInfo(issue);
105 		loadCoreMetadata(upCit, cxCit);
106 		loadXrefs(upCit, cxCit);
107 		// Special bits:
108 		switch (upCit.getCitationType()){
109 		case BOOK:
110 			Book book = (Book) upCit;
111 			load(cxCit, book);
112 			break;
113 		case PATENT:
114 			Patent patent = (Patent) upCit;
115 			load(cxCit, patent);
116 			break;
117 		/* Ignoring other cases */
118 		}
119 		return cxCit;
120 	}

122 	/**
123 	 * Loads core metadata - title, abstract, authors, year - from a UniProt
124 	 * citation into a CiteXplore citation.
125 	 * @param upCit
126 	 * @param cxCit
127 	 */
128 	private void loadCoreMetadata(Citation upCit,
129 			uk.ac.ebi.cdb.webservice.Result cxCit) {
130 		// Title:
131 		if (upCit instanceof HasTitle){
132 			cxCit.setTitle(((HasTitle) upCit).getTitle().getValue());
133 		}
134 		// Abstract:
135 		/* No info from old Citation objects (wait for citationNew package)
136 		List<CitationSummary> upAbstracts = upCit.getCitationSummary();
137 		if (upAbstracts != null && !upAbstracts.isEmpty()){
138 			cxCit.setAbstractText(upAbstracts.get(0).getValue());
139 		}
140 		*/
141 		// Authors:
142 		if (upCit instanceof HasAuthors){
143 			for (Author upAuthor: ((HasAuthors) upCit).getAuthors()){
144 				uk.ac.ebi.cdb.webservice.Authors cxAuthor =
145 						new uk.ac.ebi.cdb.webservice.Authors();
146 				cxAuthor.setFullName(upAuthor.getValue());
147 				cxCit.getAuthorList().getAuthor().add(cxAuthor);
148 			}
149 		}
150 		// Year:
151 		if (upCit instanceof HasPublicationDate){
152 			// UniProt's publication date might come as 'JUL-2010':
153 			Pattern p = Pattern.compile("(?:\\w{3}-)?(\\d{4})(?:-\\d{2}(?:-\\d{2}(?:.*)?)?)?");
154 			String upDate = ((HasPublicationDate) upCit)
155 					.getPublicationDate().getValue();
156 			Matcher m = p.matcher(upDate);
157 			if (m.matches()){
158 				cxCit.getJournalInfo().setYearOfPublication(Short.valueOf(m.group(1)));
159 			} else {
160 				LOGGER.warn("Date pattern not recognised: " + upDate);
161 			}
162 		}
163 		// Volume:
164 		if (upCit instanceof HasVolume){
165 			Volume volume = ((HasVolume) upCit).getVolume();
166 			if (volume != null){
167 				cxCit.getJournalInfo().setVolume(volume.getValue());
168 			}
169 		}
170 		// Journal name:
171 		if (upCit instanceof HasJournalName){
172 			cxCit.getJournalInfo().getJournal().setTitle(
173 					((HasJournalName) upCit).getJournalName().getValue());
174 		}
175 		// Pages:
176 		if (upCit instanceof HasPages){
177 			setPages(cxCit, ((HasPages) upCit).getFirstPage(),
178 					((HasPages) upCit).getLastPage());
179 		}
180 	}

241 	 * @param upCit
242 	 * @param cxCit

244 	private void loadXrefs(Citation upCit,
245 			uk.ac.ebi.cdb.webservice.Result cxCit) {
246 		if (upCit instanceof HasPubMedId){
247 			PubMedId pubmedId = ((HasPubMedId) upCit).getPubMedId();
248 			if (pubmedId != null){
249 				cxCit.setSource(DataSource.MED.name());
250 				cxCit.setId(pubmedId.getValue());
251 			}
252 		} else if (upCit instanceof HasAgricolaId){
253 			AgricolaId agricolaId = ((HasAgricolaId) upCit).getAgricolaId();
254 			if (agricolaId != null){
255 				cxCit.setSource(DataSource.AGR.name());
256 				cxCit.setId(agricolaId.getValue());
257 			}
258 		}

261 	private uk.ac.ebi.cdb.webservice.Result getCitationFromCitexplore(Citation upCit){
262 		uk.ac.ebi.cdb.webservice.Result cxCit = null;
263 		CitexploreWSClient citexploreClient = null;
264 		try {
265 			citexploreClient =
266 					CitexploreWSClientPool.getInstance().borrowObject();
267 			if (upCit instanceof HasPubMedId){
268 				PubMedId pubmedId = ((HasPubMedId) upCit).getPubMedId();
269 				if (pubmedId != null){
270 					String val = pubmedId.getValue();
271 					if (val != null && val.length() > 0){
272 						cxCit = citexploreClient.retrieveCitation(
273 								DataSource.MED, pubmedId.getValue());
274 					}
275 				}
276 			} else if (upCit instanceof HasAgricolaId){
277 				AgricolaId agricolaId = ((HasAgricolaId) upCit).getAgricolaId();
278 				if (agricolaId != null){
279 					String val = agricolaId.getValue();
280 					if (val != null && val.length() > 0){
281 						cxCit = citexploreClient.retrieveCitation(
282 								DataSource.AGR, agricolaId.getValue());
283 					}
284 				}
285 			}
286 		} catch (Exception e) {
287 			LOGGER.error("Unable to get citation from CiteXplore", e);
288 		} finally {
289 			if (citexploreClient != null){
290 				try {
291 					CitexploreWSClientPool.getInstance()
292 							.returnObject(citexploreClient);
293 				} catch (Exception e) {
294 					LOGGER.error("Unable to return CiteXplore client", e);
295 				}
296 			}
297 		}
298 		return cxCit;
299 	}
