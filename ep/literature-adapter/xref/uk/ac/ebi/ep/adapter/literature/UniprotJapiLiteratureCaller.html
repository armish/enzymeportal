63  	throws Exception {
64  		// CiteXplore citations:
65  		HashSet<uk.ac.ebi.cdb.webservice.Result> cxCits = null;
66  		UniProtQueryService uniProtQueryService =
67  				UniProtJAPI.factory.getUniProtQueryService();
68  		Query query = UniProtQueryBuilder
69  		        .buildIDListQuery(Arrays.asList(uniprotId));
70  		AttributeIterator<UniProtEntry> it =
71  				uniProtQueryService.getAttributes(query, "ognl:citations");
72  		if (it.hasNext()){
73  			cxCits = new HashSet<uk.ac.ebi.cdb.webservice.Result>();
74  			for (Attribute att: it){
75  				// UniProt citations:
76  				@SuppressWarnings("unchecked")
77  				List<Citation> upCits = (List<Citation>) att.getValue();
78  				for (Citation upCit :upCits){
79  					uk.ac.ebi.cdb.webservice.Result cxCit
80  					        = buildCitation(upCit);
81                      if (cxCit != null) cxCits.add(cxCit);
82  				}
83  			}
84  		} else {
85  			LOGGER.warn("No citations retrieved from " + uniprotId);
86  		}
87  		return cxCits;

90      /**
91       * Builds a citation (citeXplore model) from a UniProt citation.
92       * @param upCit the UniProt citation.
93       * @return a citation in the citeXplore model, or <code>null</code> if the
94       *      citation from UniProt lacks of a database ID (see {@link
95       *      #loadXrefs(uk.ac.ebi.kraken.interfaces.uniprot.Citation, uk.ac.ebi.cdb.webservice.Result)}).
96       */
97  	private uk.ac.ebi.cdb.webservice.Result buildCitation(Citation upCit) {
98  		uk.ac.ebi.cdb.webservice.Result cxCit =
99  				new uk.ac.ebi.cdb.webservice.Result();
100         loadXrefs(upCit, cxCit);
101         // Some UniProt citations - ex. submissions - don't have PMIDs:
102         if (cxCit.getId() == null) return null;
103 		JournalInfo issue = new JournalInfo();
104 		Journal journal = new Journal();
105 		issue.setJournal(journal);
106 		cxCit.setJournalInfo(issue);
107 		loadCoreMetadata(upCit, cxCit);
108 		// Special bits:
109 		switch (upCit.getCitationType()){
110 		case BOOK:
111 			Book book = (Book) upCit;
112 			load(cxCit, book);
113 			break;
114 		case PATENT:
115 			Patent patent = (Patent) upCit;
116 			load(cxCit, patent);
117 			break;
118 		/* Ignoring other cases */
119 		}
120 		return cxCit;
121 	}

123 	/**
124 	 * Loads core metadata - title, abstract, authors, year - from a UniProt
125 	 * citation into a CiteXplore citation.
126 	 * @param upCit
127 	 * @param cxCit
128 	 */
129 	private void loadCoreMetadata(Citation upCit,
130 			uk.ac.ebi.cdb.webservice.Result cxCit) {
131 		// Title:
132 		if (upCit instanceof HasTitle){
133 			cxCit.setTitle(((HasTitle) upCit).getTitle().getValue());
134 		}
135 		// Abstract:
136 		/* No info from old Citation objects (wait for citationNew package)
137 		List<CitationSummary> upAbstracts = upCit.getCitationSummary();
138 		if (upAbstracts != null && !upAbstracts.isEmpty()){
139 			cxCit.setAbstractText(upAbstracts.get(0).getValue());
140 		}
141 		*/
142 		// Authors:
143 		if (upCit instanceof HasAuthors){
144 		    cxCit.setAuthorList(new AuthorsList());
145 			for (Author upAuthor: ((HasAuthors) upCit).getAuthors()){
146 				uk.ac.ebi.cdb.webservice.Authors cxAuthor =
147 						new uk.ac.ebi.cdb.webservice.Authors();
148 				cxAuthor.setFullName(upAuthor.getValue());
149 				cxCit.getAuthorList().getAuthor().add(cxAuthor);
150 			}
151 		}
152 		// Year:
153 		if (upCit instanceof HasPublicationDate){
154 			// UniProt's publication date might come as 'JUL-2010':
155 			Pattern p = Pattern.compile("(?:\\w{3}-)?(\\d{4})(?:-\\d{2}(?:-\\d{2}(?:.*)?)?)?");
156 			String upDate = ((HasPublicationDate) upCit)
157 					.getPublicationDate().getValue();
158 			Matcher m = p.matcher(upDate);
159 			if (m.matches()){
160                 final String pubYear = m.group(1);
161                 cxCit.getJournalInfo().setYearOfPublication(
162                         Short.valueOf(pubYear));
163 				cxCit.setPubYear(pubYear);
164 			} else {
165 				LOGGER.warn("Date pattern not recognised: " + upDate);
166 			}
167 		}
168 		// Volume:
169 		if (upCit instanceof HasVolume){
170 			Volume volume = ((HasVolume) upCit).getVolume();
171 			if (volume != null){
172 				cxCit.getJournalInfo().setVolume(volume.getValue());
173 			}
174 		}
175 		// Journal name:
176 		if (upCit instanceof HasJournalName){
177 			cxCit.getJournalInfo().getJournal().setTitle(
178 					((HasJournalName) upCit).getJournalName().getValue());
179 		}
180 		// Pages:
181 		if (upCit instanceof HasPages){
182 			setPages(cxCit, ((HasPages) upCit).getFirstPage(),
183 					((HasPages) upCit).getLastPage());
184 		}
185 	}

246 	 * a CiteXplore citation. Currently, only PubMed and Agricola are
247 	 * supported.
248 	 * @param upCit
249 	 * @param cxCit

251 	private void loadXrefs(Citation upCit,
252 			uk.ac.ebi.cdb.webservice.Result cxCit) {
253 		if (upCit instanceof HasPubMedId){
254 			PubMedId pubmedId = ((HasPubMedId) upCit).getPubMedId();
255 			if (pubmedId != null){
256 				cxCit.setSource(DataSource.MED.name());
257 				cxCit.setId(pubmedId.getValue());
258 			}
259 		} else if (upCit instanceof HasAgricolaId){
260 			AgricolaId agricolaId = ((HasAgricolaId) upCit).getAgricolaId();
261 			if (agricolaId != null){
262 				cxCit.setSource(DataSource.AGR.name());
263 				cxCit.setId(agricolaId.getValue());
264 			}
265 		}

268 	private uk.ac.ebi.cdb.webservice.Result getCitationFromCitexplore(Citation upCit){
269 		uk.ac.ebi.cdb.webservice.Result cxCit = null;
270 		CitexploreWSClient citexploreClient = null;
271 		String citId = null;
272 		try {
273 			citexploreClient =
274 					CitexploreWSClientPool.getInstance().borrowObject();
275 			citexploreClient.setConnectTimeout(connectTimeout);
276 			citexploreClient.setReadTimeout(readTimeout);
277 			if (upCit instanceof HasPubMedId){
278 				PubMedId pubmedId = ((HasPubMedId) upCit).getPubMedId();
279 				if (pubmedId != null){
280 					citId = pubmedId.getValue();
281 					if (citId != null && citId.length() > 0){
282 						cxCit = citexploreClient.retrieveCitation(
283 								DataSource.MED, citId);
284 					}
285 				}
286 			} else if (upCit instanceof HasAgricolaId){
287 				AgricolaId agricolaId = ((HasAgricolaId) upCit).getAgricolaId();
288 				if (agricolaId != null){
289 					citId = agricolaId.getValue();
290 					if (citId != null && citId.length() > 0){
291 						cxCit = citexploreClient.retrieveCitation(
292 								DataSource.AGR, citId);
293 					}
294 				}
295 			}
296 		} catch (Exception e) {
297 			LOGGER.error("Unable to get citation from CiteXplore: " + citId, e);
298 		} finally {
299 			if (citexploreClient != null){
300 				try {
301 					CitexploreWSClientPool.getInstance()
302 							.returnObject(citexploreClient);
303 				} catch (Exception e) {
304 					LOGGER.error("Unable to return CiteXplore client", e);
305 				}
306 			}
307 		}
308 		return cxCit;
309 	}
