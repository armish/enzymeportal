49  	private void openNewSession() {
50  		session = sessionFactory.getCurrentSession();
51  		session.setFlushMode(FlushMode.COMMIT);
52  		session.beginTransaction();
53  		entryForAccessionQuery = null;
54  		xrefsForAccessionQuery = null;
55  		allXrefsForAccessionQuery =  null;
56  		xrefsForEntryQuery = null;
57  		allXrefsForEntryQuery = null;
58  	}

60  	/**
61  	 * Checks if the session has reached the chunk size to commit.
62  	 */
63  	private void checkChunkSize() {
64  		if (counter++ >= chunkSize){
65  			session.getTransaction().commit();
66  			openNewSession();
67  			counter = 0;
68  		}
69  	}

72  	 * Converts an array of database objects into an array of database names.
73  	 * @param dbs an array of {@link MmDatabase}
74  	 * @return an array of String
75  	 */
76  	private String[] getDbNames(MmDatabase[] dbs){
77  		String[] dbNames = new String[dbs.length];
78  		for (int i = 0; i < dbs.length; i++) {
79  			dbNames[i] = dbs[i].name();
80  		}
81  		return dbNames;
82  	}

84  	private String getDbNamesAsString(MmDatabase[] dbs){
85  		StringBuilder sb = new StringBuilder();
86  		for (int i = 0; i < dbs.length; i++) {
87  			if (i > 0) sb.append(',');
88  			sb.append('\'').append(dbs[i].name()).append('\'');
89  		}
90  		return sb.toString();
91  	}

93  	public void writeEntry(Entry entry) throws IOException {
94          if (entry.getEntryName() != null && entry.getEntryName().length() > 300){
95              logger.warn("[BIG: " + entry.getEntryName().length() + "] "
96              		+ entry.getEntryName());
97          }
98  		session.merge(entry); // save or saveOrUpdate does not work!
99  		logger.debug(entry.getEntryId() + " written");

199 	 * {@inheritDoc}
200 	 * <br>
201 	 * Note that this method is currently slow when the database array has more
202 	 * than one element. It is probably faster to make separate queries for
203 	 * each of the databases and then merge the results.
204 	 * (See {@link MegaDbMapperTest#testGetXrefsByAccession()})
205 	 */
206 	@SuppressWarnings("unchecked")
207 	public Collection<XRef> getXrefs(MmDatabase db, String accession,
208 			MmDatabase... xDbs) {
209 		if (xrefsForAccessionQuery == null){
210 			xrefsForAccessionQuery = session.createQuery("from XRef" +
211 					" where (fromEntry.dbName = :dbName" +
212 					" and :accession in elements(fromEntry.entryAccessions)" +
213 					" and toEntry.dbName in (:xDbNames))" +
214 					" or (toEntry.dbName = :dbName" +
215 					" and :accession in elements(toEntry.entryAccessions)" +
216 					" and fromEntry.dbName in (:xDbNames))");
217 		}
218 		return (List<XRef>) xrefsForAccessionQuery
219 				.setString("dbName", db.name())
220 				.setString("accession", accession)
221 				.setParameterList("xDbNames", getDbNames(xDbs))
222 				.list();
