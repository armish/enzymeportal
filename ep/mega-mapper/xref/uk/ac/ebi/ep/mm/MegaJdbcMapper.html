48      public void writeEntry(Entry entry) throws IOException {
49          try {
50              if (existsInMegaMap(entry)) {
51                  return;
52              }
53              PreparedStatement wEntryStm = sqlLoader.getPreparedStatement(
54                      "--insert.entry", new String[]{"id"}, (Object) null);
55              int paramNum = 1;
56  //			wEntryStm.setInt(paramNum++, entry.getId());
57              wEntryStm.setString(paramNum++, entry.getDbName());
58              wEntryStm.setString(paramNum++, entry.getEntryId());
59              if (entry.getEntryName() != null) {
60                  wEntryStm.setString(paramNum++, entry.getEntryName());
61              } else {
62                  wEntryStm.setNull(paramNum++, Types.VARCHAR);
63              }
64              wEntryStm.execute();
65              final ResultSet generatedKeys = wEntryStm.getGeneratedKeys();
66              if (generatedKeys.next()) {
67                  int id = generatedKeys.getInt(1);
68                  entry.setId(id);
69  //				Statement checkMe = generatedKeys.getStatement();
70  //				checkMe.close();
71              } else {
72                  LOGGER.warn("No generated keys!");
73              }
74              generatedKeys.close();
75              if (entry.getEntryAccessions() != null) {
76                  int index = 0;
77                  PreparedStatement wAccStm =
78                          sqlLoader.getPreparedStatement("--insert.accession");
79                  for (String accession : entry.getEntryAccessions()) {
80                      paramNum = 1;
81                      wAccStm.setInt(paramNum++, entry.getId());
82                      wAccStm.setString(paramNum++, accession);
83                      wAccStm.setInt(paramNum++, index++);
84                      wAccStm.execute();
85                  }
86              }
87          } catch (SQLException e) {
88              throw new IOException(e);
89          }
90      }

104     public void writeXref(XRef xref) throws IOException {
105         try {
106             if (existsInMegaMap(xref)) {
107                 return;
108             }
109             writeEntry(xref.getFromEntry());
110             writeEntry(xref.getToEntry());
111             PreparedStatement wXrefStm = sqlLoader.getPreparedStatement(
112                     "--insert.xref", new String[]{"id"}, (Object) null);
113             int paramNum = 1;
114 //			wXrefStm.setInt(paramNum++, xref.getId());
115             wXrefStm.setInt(paramNum++, xref.getFromEntry().getId());
116             wXrefStm.setString(paramNum++, xref.getRelationship());
117             wXrefStm.setInt(paramNum++, xref.getToEntry().getId());
118             wXrefStm.execute();
119             final ResultSet generatedKeys = wXrefStm.getGeneratedKeys();
120             if (generatedKeys.next()) {
121                 int id = generatedKeys.getInt(1);
122                 xref.setId(id);
123             } else {
124                 LOGGER.warn("No generated keys!");
125             }
126             generatedKeys.close();
127         } catch (SQLException e) {
128             LOGGER.error(xref, e);
129             throw new IOException(e);

176     /**
177      * Checks if an entry already exists in the database. If so, the passed
178      * {@link Entry} object is updated with the internal id. Additionally, if
179      * the passed entry has got a name but the one stored in the mega-map does
180      * not, the latter is updated.
181      *
182      * @param entry the entry whose existance is checked.
183      * @return <code>true</code> if the entry exists.
184      * @throws SQLException
185      */
186     private boolean existsInMegaMap(Entry entry) throws SQLException {
187 
188         PreparedStatement rEntryStm =
189                 sqlLoader.getPreparedStatement("--entry.by.entryid");
190         int paramNum = 1;
191         rEntryStm.setString(paramNum++, entry.getEntryId());
192         rEntryStm.setString(paramNum++, entry.getDbName());
193         final ResultSet rs = rEntryStm.executeQuery();
194         final boolean exists = rs.next();
195         if (exists) {
196             entry.setId(rs.getInt("id"));
197             // Check whether the stored name is the same:
198             String nameInDb = rs.getString("entry_name");
199             if (nameInDb == null && entry.getEntryName() != null) {
200                 updateEntry(entry);
201                 LOGGER.info(MessageFormat.format(
202                         "Added name {0} to {1} entry {2}",
203                         entry.getEntryName(), entry.getDbName(),
204                         entry.getEntryId()));
205             } else if (nameInDb != null && entry.getEntryName() != null
206                     && !nameInDb.equals(entry.getEntryName())) {
207                 LOGGER.warn(MessageFormat.format(
208                         "{0} entry {1} exists with a name {2} != {3}",
209                         entry.getDbName(), entry.getEntryId(), nameInDb,
210                         entry.getEntryName()));
211             }
212         }
213         rs.close();
214         return exists;
215     }

272     /**
273      * Builds a list of XRef objects from a result set, and closes it.
274      *
275      * @param rs
276      * @return a list of {@link XRef}s, or <code>null</code> if none found.
277      * @throws SQLException
278      */
279     private List<XRef> buildXref(ResultSet rs) throws SQLException {
280         List<XRef> xrefs = null;
281         while (rs.next()) {
282             if (xrefs == null) {
283                 xrefs = new ArrayList<XRef>();
284             }
285             XRef xref = new XRef();
286             xref.setId(rs.getInt("id"));
287             xref.setFromEntry(getEntryById(rs.getInt("from_entry")));
288             xref.setToEntry(getEntryById(rs.getInt("to_entry")));
289             xref.setRelationship(rs.getString("relationship"));
290             xrefs.add(xref);
291         }
292         rs.close();
293         return xrefs;
294     }

500     /**
501      * Converts an enumerated value into a SQLLoader constraint key.
502      *
503      * @param constraint the constraint to apply.
504      * @return the key to be used by the SQLLoader.
505      */
506     private String getSqlConstraint(Constraint constraint) {
507         switch (constraint) {
508             case EQUALS:
509                 return "--constraint.equals";
510             case STARTS_WITH:
511             case CONTAINS:
512             case ENDS_WITH:
513                 return "--constraint.like";
514             default:
515                 throw new IllegalArgumentException("constraint not supported: "
516                         + constraint);
517         }
518     }

520     /**
521      * Builds the appropriate oracle query string for the constraint.
522      *
523      * @param constraint a constraint.
524      * @param idFragment a fragment of the string to match.
525      * @return a string with the wildcard % in the appropriate place(s).
526      */
527     private String getIdParameter(Constraint constraint, String idFragment) {
528         switch (constraint) {
529             case EQUALS:
530                 return idFragment;
531             case STARTS_WITH:
532                 return idFragment.replace("_", "\\_") + "%";
533             case CONTAINS:
534                 return "%" + idFragment.replace("_", "\\_") + "%";
535             case ENDS_WITH:
536                 return "%" + idFragment.replace("_", "\\_");
537             default:
538                 throw new IllegalArgumentException("constraint not supported: "
539                         + constraint);
540         }
541     }

1129         DecimalFormatSymbols currentLocaleSymbols = DecimalFormatSymbols.getInstance();
1130         char localeMinusSign = currentLocaleSymbols.getMinusSign();
1131 
1132         if (!Character.isDigit(str.charAt(0)) && str.charAt(0) != localeMinusSign) {
1133             return false;
1134         }
1135 
1136         boolean isDecimalSeparatorFound = false;
1137         char localeDecimalSeparator = currentLocaleSymbols.getDecimalSeparator();
1138 
1139         for (char c : str.substring(1).toCharArray()) {
1140             if (!Character.isDigit(c)) {
1141                 if (c == localeDecimalSeparator && !isDecimalSeparatorFound) {
1142                     isDecimalSeparatorFound = true;
1143                     continue;
1144                 }
1145                 return false;
1146             }
1147         }
1148         return true;
1149     }
