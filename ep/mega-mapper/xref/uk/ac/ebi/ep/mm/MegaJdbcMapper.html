39      public void writeEntry(Entry entry) throws IOException {
40          try {
41              if (existsInMegaMap(entry)) {
42                  return;
43              }
44              PreparedStatement wEntryStm = sqlLoader.getPreparedStatement(
45                      "--insert.entry", new String[]{"id"}, (Object) null);
46              int paramNum = 1;
47  //			wEntryStm.setInt(paramNum++, entry.getId());
48              wEntryStm.setString(paramNum++, entry.getDbName());
49              wEntryStm.setString(paramNum++, entry.getEntryId());
50              if (entry.getEntryName() != null) {
51                  wEntryStm.setString(paramNum++, entry.getEntryName());
52              } else {
53                  wEntryStm.setNull(paramNum++, Types.VARCHAR);
54              }
55              wEntryStm.execute();
56              final ResultSet generatedKeys = wEntryStm.getGeneratedKeys();
57              if (generatedKeys.next()) {
58                  int id = generatedKeys.getInt(1);
59                  entry.setId(id);
60  //				Statement checkMe = generatedKeys.getStatement();
61  //				checkMe.close();
62              } else {
63                  LOGGER.warn("No generated keys!");
64              }
65              generatedKeys.close();
66              if (entry.getEntryAccessions() != null) {
67                  int index = 0;
68                  PreparedStatement wAccStm =
69                          sqlLoader.getPreparedStatement("--insert.accession");
70                  for (String accession : entry.getEntryAccessions()) {
71                      paramNum = 1;
72                      wAccStm.setInt(paramNum++, entry.getId());
73                      wAccStm.setString(paramNum++, accession);
74                      wAccStm.setInt(paramNum++, index++);
75                      wAccStm.execute();
76                  }
77              }
78          } catch (SQLException e) {
79              throw new IOException(e);
80          }
81      }

92       * {@inheritDoc} <br> This implementation writes any (or both) of the two
93       * linked entries in case they don't already exist in the database.

95      public void writeXref(XRef xref) throws IOException {
96          try {
97              if (existsInMegaMap(xref)) {
98                  return;
99              }
100             writeEntry(xref.getFromEntry());
101             writeEntry(xref.getToEntry());
102             PreparedStatement wXrefStm = sqlLoader.getPreparedStatement(
103                     "--insert.xref", new String[]{"id"}, (Object) null);
104             int paramNum = 1;
105 //			wXrefStm.setInt(paramNum++, xref.getId());
106             wXrefStm.setInt(paramNum++, xref.getFromEntry().getId());
107             wXrefStm.setString(paramNum++, xref.getRelationship());
108             wXrefStm.setInt(paramNum++, xref.getToEntry().getId());
109             wXrefStm.execute();
110             final ResultSet generatedKeys = wXrefStm.getGeneratedKeys();
111             if (generatedKeys.next()) {
112                 int id = generatedKeys.getInt(1);
113                 xref.setId(id);
114             } else {
115                 LOGGER.warn("No generated keys!");
116             }
117             generatedKeys.close();
118         } catch (SQLException e) {
119             LOGGER.error(xref, e);
120             throw new IOException(e);
121         }
122     }

167     /**
168      * Checks if an entry already exists in the database. If so, the passed {@link Entry}
169      * object is updated with the internal id.
170      *
171      * @param entry
172      * @return
173      * <code>true</code> if the entry exists.
174      * @throws SQLException
175      */
176     private boolean existsInMegaMap(Entry entry) throws SQLException {
177 
178         PreparedStatement rEntryStm =
179                 sqlLoader.getPreparedStatement("--entry.by.entryid");
180         int paramNum = 1;
181         rEntryStm.setString(paramNum++, entry.getEntryId());
182         rEntryStm.setString(paramNum++, entry.getDbName());
183         final ResultSet rs = rEntryStm.executeQuery();
184         final boolean exists = rs.next();
185         if (exists) {
186             entry.setId(rs.getInt("id"));
187         }
188         rs.close();
189         return exists;
190     }

192     private boolean existsInMegaMap(XRef xref) throws SQLException {
193         PreparedStatement rXrefStm =
194                 sqlLoader.getPreparedStatement("--xref.by.entryids");
195         int paramNum = 1;
196         rXrefStm.setString(paramNum++, xref.getFromEntry().getDbName());
197         rXrefStm.setString(paramNum++, xref.getFromEntry().getEntryId());
198         rXrefStm.setString(paramNum++, xref.getToEntry().getDbName());
199         rXrefStm.setString(paramNum++, xref.getToEntry().getEntryId());
200         final ResultSet rs = rXrefStm.executeQuery();
201         final boolean exists = rs.next();
202         if (exists) {
203             final int xrefId = rs.getInt("id");
204             LOGGER.warn("XRef already exists in the database: "
205                     + xref.toString() + " as " + xrefId);
206             xref.setId(xrefId);
207         }
208         rs.close();
209         return exists;
210     }

247     /**
248      * Builds a list of XRef objects from a result set, and closes it.
249      *
250      * @param rs
251      * @return a list of {@link XRef}s, or
252      * <code>null</code> if none found.
253      * @throws SQLException
254      */
255     private List<XRef> buildXref(ResultSet rs) throws SQLException {
256         List<XRef> xrefs = null;
257         while (rs.next()) {
258             if (xrefs == null) {
259                 xrefs = new ArrayList<XRef>();
260             }
261             XRef xref = new XRef();
262             xref.setId(rs.getInt("id"));
263             xref.setFromEntry(getEntryById(rs.getInt("from_entry")));
264             xref.setToEntry(getEntryById(rs.getInt("to_entry")));
265             xref.setRelationship(rs.getString("relationship"));
266             xrefs.add(xref);
267         }
268         rs.close();
269         return xrefs;
270     }

307     public Collection<XRef> getXrefs(Entry entry) {
308         Collection<XRef> xrefs = null;
309         try {
310             if (entry.getId() == null) {
311                 // update with value from the database:
312                 existsInMegaMap(entry);
313             }
314             if (entry.getId() != null) {
315                 PreparedStatement allXrefsByEntryStm =
316                         sqlLoader.getPreparedStatement("--xrefs.all.by.entry");
317                 allXrefsByEntryStm.setInt(1, entry.getId());
318                 allXrefsByEntryStm.setInt(2, entry.getId());
319                 ResultSet rs = allXrefsByEntryStm.executeQuery();
320                 xrefs = buildXref(rs);
321             } else {
322                 LOGGER.warn(entry.toString() + " not known in the mega-map");
323             }
324         } catch (SQLException e) {
325             LOGGER.error(entry.getId(), e);
326         }
327         return xrefs;
328     }

415     /**
416      * {@inheritDoc}
417      * <br>Please note that this implementation uses backslash (\) as escape
418      * character for any underscore (_) present in the idFragment, so that it
419      * is taken as part of the ID instead of an oracle wildcard.
420      */
421 	public Collection<XRef> getXrefs(MmDatabase db, String idFragment,
422 			Constraint constraint, Relationship rel) {
423 		Collection<XRef> xrefs = null;
424 		String idConst = getSqlConstraint(constraint);
425 		idFragment = getIdParameter(constraint, idFragment);
426 		String dbConst = db == null? "" : "--constraint.db";
427 		String relConst = rel == null? "" : "--constraint.relationship";
428 		try {
429 			PreparedStatement ps = sqlLoader.getPreparedStatement(
430 					"--xrefs.by.id.fragment", idConst, dbConst, relConst);
431 			int n = 1;
432 			ps.setString(n++, idFragment);
433 			if (db != null) ps.setString(n++, db.name());
434 			ps.setString(n++, idFragment);
435 			if (db != null) ps.setString(n++, db.name());
436 			if (rel != null) ps.setString(n++, rel.name());
437             xrefs = buildXref(ps.executeQuery());
438 		} catch (SQLException e) {
439 			LOGGER.error(db + " - " + idFragment + " - " + constraint
440 					+ " - " + rel, e);
441 		}
442 		return xrefs;
443 	}

446 			Constraint constraint, MmDatabase... xDbs) {
447 		Collection<XRef> xrefs = null;
448 		String idConst = getSqlConstraint(constraint);
449 		idFragment = getIdParameter(constraint, idFragment);
450 		String dbConst = db == null? "" : "--constraint.db";
451 		String inClause = dbArrayForQuery(xDbs);
452 		try {
453 			PreparedStatement ps = sqlLoader.getPreparedStatement(
454 					"--xrefs.by.id.fragment.and.db",
455 					idConst, dbConst, inClause);
456 			int n = 1;
457 			ps.setString(n++, idFragment);
458 			if (db != null) ps.setString(n++, db.name());
459 			xrefs = buildXref(ps.executeQuery());
460 		} catch (SQLException e) {
461 			LOGGER.error(db + " - " + idFragment + " - " + constraint
462 					+ " - " + xDbs, e);
463 		}
464 		return xrefs;

467 	/**
468 	 * Converts an enumerated value into a SQLLoader constraint key.
469 	 * @param constraint the constraint to apply.
470 	 * @return the key to be used by the SQLLoader.
471 	 */
472 	private String getSqlConstraint(Constraint constraint){
473 		switch (constraint) {
474 		case EQUALS:
475 			return "--constraint.equals";
476 		case STARTS_WITH:
477 		case CONTAINS:
478 		case ENDS_WITH:
479 			return "--constraint.like";
480 		default:
481 			throw new IllegalArgumentException("constraint not supported: "
482 					+ constraint);
483 		}
484 	}
485 	
486 	/**
487 	 * Builds the appropriate oracle query string for the constraint.
488 	 * @param constraint a constraint.
489 	 * @param idFragment a fragment of the string to match.
490 	 * @return a string with the wildcard % in the appropriate place(s).
491 	 */
492 	private String getIdParameter(Constraint constraint, String idFragment){
493 		switch (constraint) {
494 		case EQUALS:
495 			return idFragment;
496 		case STARTS_WITH:
497 			return idFragment.replace("_", "\\_") + "%";
498 		case CONTAINS:
499 			return "%" + idFragment.replace("_", "\\_") + "%";
500 		case ENDS_WITH:
501 			return "%" + idFragment.replace("_", "\\_");
502 		default:
503 			throw new IllegalArgumentException("constraint not supported: "
504 					+ constraint);
505 		}
506 	}
