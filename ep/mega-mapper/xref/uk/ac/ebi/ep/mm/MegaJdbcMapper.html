42      public void writeEntry(Entry entry) throws IOException {
43          try {
44              if (existsInMegaMap(entry)) {
45                  return;
46              }
47              PreparedStatement wEntryStm = sqlLoader.getPreparedStatement(
48                      "--insert.entry", new String[]{"id"}, (Object) null);
49              int paramNum = 1;
50  //			wEntryStm.setInt(paramNum++, entry.getId());
51              wEntryStm.setString(paramNum++, entry.getDbName());
52              wEntryStm.setString(paramNum++, entry.getEntryId());
53              if (entry.getEntryName() != null) {
54                  wEntryStm.setString(paramNum++, entry.getEntryName());
55              } else {
56                  wEntryStm.setNull(paramNum++, Types.VARCHAR);
57              }
58              wEntryStm.execute();
59              final ResultSet generatedKeys = wEntryStm.getGeneratedKeys();
60              if (generatedKeys.next()) {
61                  int id = generatedKeys.getInt(1);
62                  entry.setId(id);
63  //				Statement checkMe = generatedKeys.getStatement();
64  //				checkMe.close();
65              } else {
66                  LOGGER.warn("No generated keys!");
67              }
68              generatedKeys.close();
69              if (entry.getEntryAccessions() != null) {
70                  int index = 0;
71                  PreparedStatement wAccStm =
72                          sqlLoader.getPreparedStatement("--insert.accession");
73                  for (String accession : entry.getEntryAccessions()) {
74                      paramNum = 1;
75                      wAccStm.setInt(paramNum++, entry.getId());
76                      wAccStm.setString(paramNum++, accession);
77                      wAccStm.setInt(paramNum++, index++);
78                      wAccStm.execute();
79                  }
80              }
81          } catch (SQLException e) {
82              throw new IOException(e);
83          }
84      }

98      public void writeXref(XRef xref) throws IOException {
99          try {
100             if (existsInMegaMap(xref)) {
101                 return;
102             }
103             writeEntry(xref.getFromEntry());
104             writeEntry(xref.getToEntry());
105             PreparedStatement wXrefStm = sqlLoader.getPreparedStatement(
106                     "--insert.xref", new String[]{"id"}, (Object) null);
107             int paramNum = 1;
108 //			wXrefStm.setInt(paramNum++, xref.getId());
109             wXrefStm.setInt(paramNum++, xref.getFromEntry().getId());
110             wXrefStm.setString(paramNum++, xref.getRelationship());
111             wXrefStm.setInt(paramNum++, xref.getToEntry().getId());
112             wXrefStm.execute();
113             final ResultSet generatedKeys = wXrefStm.getGeneratedKeys();
114             if (generatedKeys.next()) {
115                 int id = generatedKeys.getInt(1);
116                 xref.setId(id);
117             } else {
118                 LOGGER.warn("No generated keys!");
119             }
120             generatedKeys.close();
121         } catch (SQLException e) {
122             LOGGER.error(xref, e);
123             throw new IOException(e);
124         }
125     }

194     private boolean existsInMegaMap(XRef xref) throws SQLException {
195         PreparedStatement rXrefStm =
196                 sqlLoader.getPreparedStatement("--xref.by.entryids");
197         int paramNum = 1;
198         rXrefStm.setString(paramNum++, xref.getFromEntry().getDbName());
199         rXrefStm.setString(paramNum++, xref.getFromEntry().getEntryId());
200         rXrefStm.setString(paramNum++, xref.getToEntry().getDbName());
201         rXrefStm.setString(paramNum++, xref.getToEntry().getEntryId());
202         final ResultSet rs = rXrefStm.executeQuery();
203         final boolean exists = rs.next();
204         if (exists) {
205             final int xrefId = rs.getInt("id");
206             LOGGER.warn("XRef already exists in the database: "
207                     + xref.toString() + " as " + xrefId);
208             xref.setId(xrefId);
209         }
210         rs.close();
211         return exists;

249     /**
250      * Builds a list of XRef objects from a result set, and closes it.
251      *
252      * @param rs
253      * @return a list of {@link XRef}s, or <code>null</code> if none found.
254      * @throws SQLException
255      */
256     private List<XRef> buildXref(ResultSet rs) throws SQLException {
257         List<XRef> xrefs = null;
258         while (rs.next()) {
259             if (xrefs == null) {
260                 xrefs = new ArrayList<XRef>();
261             }
262             XRef xref = new XRef();
263             xref.setId(rs.getInt("id"));
264             xref.setFromEntry(getEntryById(rs.getInt("from_entry")));
265             xref.setToEntry(getEntryById(rs.getInt("to_entry")));
266             xref.setRelationship(rs.getString("relationship"));
267             xrefs.add(xref);
268         }
269         rs.close();
270         return xrefs;
271     }

312                 // update with value from the database:
313                 existsInMegaMap(entry);
314             }
315             if (entry.getId() != null) {
316                 PreparedStatement allXrefsByEntryStm =
317                         sqlLoader.getPreparedStatement("--xrefs.all.by.entry");
318                 allXrefsByEntryStm.setInt(1, entry.getId());
319                 allXrefsByEntryStm.setInt(2, entry.getId());
320                 ResultSet rs = allXrefsByEntryStm.executeQuery();
321                 xrefs = buildXref(rs);
322             } else {
323                 LOGGER.warn(entry.toString() + " not known in the mega-map");
324             }
325         } catch (SQLException e) {
326             LOGGER.error(entry.getId(), e);
327         }
328         return xrefs;

476     /**
477      * Converts an enumerated value into a SQLLoader constraint key.
478      *
479      * @param constraint the constraint to apply.
480      * @return the key to be used by the SQLLoader.
481      */
482     private String getSqlConstraint(Constraint constraint) {
483         switch (constraint) {
484             case EQUALS:
485                 return "--constraint.equals";
486             case STARTS_WITH:
487             case CONTAINS:
488             case ENDS_WITH:
489                 return "--constraint.like";
490             default:
491                 throw new IllegalArgumentException("constraint not supported: "
492                         + constraint);
493         }
494     }

496     /**
497      * Builds the appropriate oracle query string for the constraint.
498      *
499      * @param constraint a constraint.
500      * @param idFragment a fragment of the string to match.
501      * @return a string with the wildcard % in the appropriate place(s).
502      */
503     private String getIdParameter(Constraint constraint, String idFragment) {
504         switch (constraint) {
505             case EQUALS:
506                 return idFragment;
507             case STARTS_WITH:
508                 return idFragment.replace("_", "\\_") + "%";
509             case CONTAINS:
510                 return "%" + idFragment.replace("_", "\\_") + "%";
511             case ENDS_WITH:
512                 return "%" + idFragment.replace("_", "\\_");
513             default:
514                 throw new IllegalArgumentException("constraint not supported: "
515                         + constraint);
516         }
517     }
