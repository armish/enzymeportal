14  	 * Processes the command line options.
15  	 * @param args (order does not matter):
16  	 * 		-xmlFile &lt;xmlFile&gt; (-indexDir &lt;indexDir&gt;|-dbConfig
17  	 * 		&lt;dbConfig&gt;)
18  	 * <ul>
19  	 * 	<li>-xmlFile &lt;xmlFile&gt;: the XML (or HTML) file to parse</li>
20  	 * 	<li>-indexDir &lt;indexDir&gt; (optional): the directory for a lucene
21  	 * 		index. If it does not exist, a new one is created.</li>
22  	 * 	<li>-dbConfig &lt;dbConfig&gt; (optional): the name of a database
23  	 * 		configuration (hibernate configuration file).</li>
24  	 * </ul>
25  	 * Either <code>-indexDir</code> or <code>-dbConfig</code> must be defined,
26  	 * but not both. If none of them is defined, the default hibernate config
27  	 * file - <code>hibernate.cfg.xml</code> - will be used as dbConfig. 
28  	 * @return a commandLine object with the required options, or
29  	 * 		<code>null</code> if the arguments are not valid.
30  	 */
31  	@SuppressWarnings("static-access")
32  	static CommandLine getCommandLine(String... args) {
33  		Options options = new Options();
34          options.addOption(OptionBuilder.isRequired()
35                  .hasArg().withArgName("xmlFile")
36                  .withDescription("XML (or HTML) file")
37                  .create("xmlFile"));
38          options.addOption(OptionBuilder
39                  .hasArg().withArgName("indexDir")
40                  .withDescription("Lucene index directory")
41                  .create("indexDir"));
42          options.addOption(OptionBuilder
43                  .hasArg().withArgName("dbConfig")
44                  .withDescription("Configuration for DB connection")
45                  .create("dbConfig"));
46          CommandLine cl = null;
47          try {
48              cl = new GnuParser().parse(options, args);
49  	        if (cl.hasOption("indexDir") && cl.hasOption("dbCOnfig")){
50              	throw new ParseException("Either indexDir or dbConfig must be" +
51              			" defined, but not both");
52  	        }
53          } catch (ParseException e){
54              new HelpFormatter().printHelp(MmParser.class.getName(), options);
55          }
56  		return cl;
57  	}
