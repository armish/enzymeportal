93  	public void parse(String xmlFile)
94  	throws Exception {
95  		if (mm == null){
96  			// Don't go ahead:
97  			throw new NullPointerException("A MegaMapper must be configured");
98  		}
99  		try {
100 			mm.openMap();
101 			LOGGER.info("Parsing start");
102 			EntriesType entries = getEntries(xmlFile);
103 			for (EntryType entry : entries.getEntry()) {
104 				// no cross-references? No use:
105 				if (entry.getCrossReferences() == null) continue;
106 				Entry ent = new Entry();
107 				ent.setEntryId(entry.getId());
108 				ent.setEntryName(entry.getName());
109 				ent.setEntryAccessions(Collections.singletonList(entry.getAcc()));
110 				ent.setDbName(database.name());
111 				mm.writeEntry(ent);
112 				
113 				for (RefType xref : entry.getCrossReferences().getRef()) {
114 					/*
115 					*/
116 				}
117 				List<String> uniprotAccs = new ArrayList<String>();
118 				for (RefType xref : entry.getCrossReferences().getRef()) {
119 					// Take only UniProt:
120 					if (MmDatabase.UniProt.equals(MmDatabase.parse(xref.getDbname()))){
121 						uniprotAccs.add(xref.getDbkey());
122 					}
123 				}
124 				if (!uniprotAccs.isEmpty()){
125 					// We may associate with the accession if it is in the index.
126 				    // We might have too many xrefs for lucene to handle (max. is
127 					// 1024 by default), let's pass them in chunks:
128                     int chunkSize = 1024;
129                     for (int i = 0; i < ((uniprotAccs.size() % chunkSize) + 1); i++){
130                         int from = i * chunkSize;
131                         int to = from + chunkSize - 1;
132                         if (to > uniprotAccs.size()) to = uniprotAccs.size();
133                         List<String> chunk = uniprotAccs.subList(from, to); 
134 
135                         Query query = queryParser.parse(getQuery(MmDatabase.UniProt, chunk));
136                         ScoreDoc[] hits = indexSearcher.search(query, null, 1000).scoreDocs;
137                         if (hits.length > 0){
138                             List<Document> hitDocs = new ArrayList<Document>();
139                             for (int j = 0; j < hits.length; j++) {
140                                 Document hitDoc = indexSearcher.doc(hits[j].doc);
141                                 hitDoc.add(new Field(
142                                 		database.getIdField(), ent.getEntryId(),
143                                         Field.Store.YES, Field.Index.ANALYZED));
144                                 hitDoc.add(new Field(
145                                 		database.getNameField(), ent.getEntryName(),
146                                         Field.Store.YES, Field.Index.ANALYZED));
147                                 hitDocs.add(hitDoc);
148                             }
149                             // Write changes
150                             indexWriter.deleteDocuments(query);
151                             for (Document hitDoc : hitDocs) {
152                                 indexWriter.addDocument(hitDoc);
153                             }
154                         }
155 
156                         if (to == uniprotAccs.size()) break;
157                     }
158 				}
159 			}
160 			LOGGER.info("Parsing end");
161         } catch (Exception e){
162             LOGGER.error("During parsing", e);
163             indexWriter.rollback();
164             throw e;
165 		} finally {
166 			mm.closeMap();
167 		}
168 	}

185 	/**
186 	 * Detects the database being processed - stored in the field
187 	 * <code>dbField</code> - and returns the entries in it.
188 	 * @param xmlFile an EB-Eye XML file.
189 	 * @return the entries in the XML file.
190 	 * @throws JAXBException
191 	 * @throws FileNotFoundException
192 	 * @throws IOException
193 	 */
194 	private EntriesType getEntries(String xmlFile)
195 	throws JAXBException, FileNotFoundException, IOException {
196 		LOGGER.info("Getting entries from EB-Eye XML file.");
197 		JAXBContext jc = JAXBContext.newInstance( "uk.ac.ebi.ebinocle" );
198 		Unmarshaller u = jc.createUnmarshaller();
199 		FileInputStream fis = null;
200 		EntriesType entries = null;
201 		try {
202 			fis = new FileInputStream(xmlFile);
203 			@SuppressWarnings("unchecked")
204 			JAXBElement<DatabaseType> elem = (JAXBElement<DatabaseType>) u.unmarshal(fis);
205 			DatabaseType db = (DatabaseType) elem.getValue();
206 			String dbName = db.getName();
207 			LOGGER.info("Getting entries for " + dbName);
208 			if (dbName.toUpperCase().startsWith("CHEBI")){
209 				database = MmDatabase.ChEBI;
210 //			} else if (dbName.toUpperCase().startsWith("CHEMBL")){
211 //				database = MmDatabase.ChEMBL;
212 			} else {
213 				throw new RuntimeException("Database not supported: " + dbName);
214 			}
215 			entries = db.getEntries();
216 			LOGGER.info("Got them!");
217 		} catch (FileNotFoundException e) {
218 			LOGGER.error("XML file not found", e);
219 			throw e;
220 		} finally {
221 			if (fis != null) fis.close();
222 		}
223 		return entries;
224 	}
