17  /**
18   * Parser for IntEnzXML which extracts data interesting for the Enzyme Portal
19   * mega-map, namely cofactors, reactions, reactants and products as
20   * cross-references:
21   * <ul>
22   *  <li>EC number &rarr; Rhea reactions</li>
23   *  <li>EC number &rarr; ChEBI entities (cofactors, substrates, products)</li>
24   *  <li>Rhea reaction &rarr; ChEBI entities (reactants, products)</li>
25   * </ul>
26   * Please note that in case of <b>reversible</b> (bidirectional) reactions -
27   * most of the enzyme classification - the relationship for ChEBI entities will
28   * be set
29   * to {@link Relationship.is_substrate_or_product_of} (for EC cross-reference)
30   * or {@link Relationship.is_reactant_or_product_of} (for Rhea cross-reference),
31   * while <b>irreversible</b> (unidirectional: left-to-right or right-to-left in
32   * Rhea speech) will have the more specific
33   * {@link Relationship.is_substrate_of}/{@link Relationship.is_product_of}
34   * (for EC cross-reference) or
35   * {@link Relationship.is_reactant_of}/{@link Relationship.is_product_of}
36   * (for Rhea cross-reference).
37   * @author rafa
38   * @since 1.0.16
39   */
40  public class IntenzSaxParser extends MmSaxParser {

82      /**
83       * Parses an IntEnzXML file and stores cross-references from EC numbers to
84       * cofactors, reactants, products and reactions into a mega-map.
85       * @param args see {@link CliOptionsParser#getCommandLine(String...)}.
86       * @throws Exception in case of error while parsing.
87       */
88      public static void main(String... args) throws Exception {
89          CommandLine cl = CliOptionsParser.getCommandLine(args);
90          if (cl != null) mainParse(cl, new IntenzSaxParser());

93      @Override
94      public void startElement(String uri, String localName, String qName,
95              Attributes attributes)
96      throws SAXException {
97          currentContext.push(qName.isEmpty()? localName : qName);
98  
99          // Update flags: todo
100         String currentXpath = getCurrentXpath();
101         isEnzyme = ENZYME.equals(currentXpath);
102         isEc = EC.equals(currentXpath);
103         isAcceptedName = ACCEPTED_NAME.equals(currentXpath);
104         isCmlReaction = CML_REACTION.equals(currentXpath);
105         isReactionName = REACTION_NAME.equals(currentXpath);
106         isRheaId = RHEA_ID.equals(currentXpath);
107         isReactant = REACTANT.equals(currentXpath);
108         isProduct = PRODUCT.equals(currentXpath);
109         if (isCmlReaction){
110             isReversibleReaction = attributes.getValue("", "convention")
111                     .contains("direction.BI");
112         }
113         isCofactor = COFACTOR.equals(currentXpath);
114 
115         // Clear placeholder:
116         if (currentChars.length() > 0){
117             currentChars.delete(0, currentChars.length());
118         }
119 
120         if (isEnzyme){
121             ecEntry = new Entry();
122             ecEntry.setDbName(MmDatabase.EC.name());
123         } else if (isCmlReaction){
124             rheaEntry = new Entry();
125             rheaEntry.setDbName(MmDatabase.Rhea.name());
126         } else if (isRheaId){
127             rheaEntry.setEntryId(attributes.getValue("", "value"));
128         } else if (isReactant){
129             reactants.add(createChebiEntry(attributes));
130         } else if (isProduct){
131             products.add(createChebiEntry(attributes));
132         } else if (isCofactor){
133             Entry cofactor = new Entry();
134             cofactor.setDbName(MmDatabase.ChEBI.name());
135             cofactor.setEntryId(attributes.getValue("", "accession"));
136             cofactors.add(cofactor);
137         }
138     }

140     private Entry createChebiEntry(Attributes attributes) {
141         Entry chebiEntry = new Entry();
142         chebiEntry.setDbName(MmDatabase.ChEBI.name());
143         chebiEntry.setEntryId(attributes.getValue("", "id"));
144         chebiEntry.setEntryName(attributes.getValue("", "title"));

157     @Override
158     public void endElement(String uri, String localName, String qName)
159     throws SAXException {
160         if (isEc){
161             ecEntry.setEntryId(currentChars.toString().replace("EC ", ""));
162         } else if (isAcceptedName){
163             ecEntry.setEntryName(currentChars.toString());
164         } else if (isReactionName){
165             rheaEntry.setEntryName(currentChars.toString());
166         } else if (isCmlReaction){
167             // Cross-reference EC-Rhea:
168             XRef ecRheaXref = new XRef();
169             ecRheaXref.setFromEntry(ecEntry);
170             ecRheaXref.setToEntry(rheaEntry);
171             ecRheaXref.setRelationship(Relationship.between(
172                     MmDatabase.EC, MmDatabase.Rhea).name());
173             xrefs.add(ecRheaXref);
174             // Cross-references ChEBI-Rhea and ChEBI-EC:
175             for (Entry reactant : reactants) {
176                 XRef chebiRheaXref = new XRef();
177                 chebiRheaXref.setFromEntry(reactant);
178                 chebiRheaXref.setToEntry(rheaEntry);
179                 chebiRheaXref.setRelationship(isReversibleReaction?
180                     Relationship.is_reactant_or_product_of.name():
181                     Relationship.is_reactant_of.name());
182                 xrefs.add(chebiRheaXref);
183                 XRef chebiEcXref = new XRef();
184                 chebiEcXref.setFromEntry(reactant);
185                 chebiEcXref.setToEntry(ecEntry);
186                 chebiEcXref.setRelationship(isReversibleReaction?
187                         Relationship.is_substrate_or_product_of.name():
188                         Relationship.is_substrate_of.name());
189                 xrefs.add(chebiEcXref);
190             }
191             for (Entry product : products) {
192                 XRef chebiRheaXref = new XRef();
193                 chebiRheaXref.setFromEntry(product);
194                 chebiRheaXref.setToEntry(rheaEntry);
195                 chebiRheaXref.setRelationship(isReversibleReaction?
196                         Relationship.is_reactant_or_product_of.name():
197                         Relationship.is_product_of.name());
198                 xrefs.add(chebiRheaXref);
199                 XRef chebiEcXref = new XRef();
200                 chebiEcXref.setFromEntry(product);
201                 chebiEcXref.setToEntry(ecEntry);
202                 chebiEcXref.setRelationship(isReversibleReaction?
203                         Relationship.is_substrate_or_product_of.name():
204                         Relationship.is_product_of.name());
205                 xrefs.add(chebiEcXref);
206             }
207             // Clean up:
208             rheaEntry = null;
209             reactants.clear();
210             products.clear();
211         } else if (isCofactor){
212             // Add the missing name:
213             cofactors.get(cofactors.size()-1)
214                     .setEntryName(currentChars.toString());
215         } else if (isEnzyme){
216             for (Entry cofactor : cofactors) {
217                 XRef xref = new XRef();
218                 xref.setFromEntry(cofactor);
219                 xref.setToEntry(ecEntry);
220                 xref.setRelationship(Relationship.is_cofactor_of.name());
221                 xrefs.add(xref);
222             }
223             if (!xrefs.isEmpty()) try {
224                 mm.writeXrefs(xrefs);
225                 // Update enzyme name from EC:
226                 mm.updateEntry(ecEntry);
227             } catch (IOException e) {
228                 throw new RuntimeException("Unable to add xrefs", e);
229             }
230             // Clean up:
231             xrefs.clear();
232             ecEntry = null;
233             cofactors.clear();
234         }
235 
236         currentContext.pop();
237         // Update flags:
238         String currentXpath = getCurrentXpath();
239         isEnzyme = ENZYME.equals(currentXpath);
240         isEc = false;
241         isAcceptedName = false;
242         isCmlReaction = CML_REACTION.equals(currentXpath);
243         isReactionName = false;
244         isRheaId = false;
245         isReactant = false;
246         isProduct = false;
247         isCofactor = false;
248     }
