101 	public void parse(String file) throws Exception {
102 		// Check the extension of the file:
103 		Format format = Format.valueOf(file.substring(file.lastIndexOf('.')+1));
104 		BufferedReader br = null;
105 		InputStreamReader isr = null;
106 		InputStream is = null;
107 		try {
108 			mm.openMap();
109 			is = file.startsWith("http://")?
110 					new URL(file).openStream():
111 					new FileInputStream(file);
112 			isr = new InputStreamReader(is);
113 			br = new BufferedReader(isr);
114 			LOGGER.info("Parsing start");
115 			String line;
116 			while ((line = br.readLine()) != null){
117 				String[] fields = getFields(format, line);
118 				if (fields == null) continue; // header lines
119 				
120 				// Start with the scores, discard immediately if not
121 				// interesting, otherwise keep for the MeSH terms (see below):
122 				String[] scoresCell = fields[4].split(" ?/ ?");
123 				double[] scores = new double[scoresCell.length];
124 				boolean interesting = false;
125 				for (int i = 0; i < scoresCell.length; i++) {
126 					final String scoreString = scoresCell[i].trim();
127 					if (scoreString.equals("exact")){
128 						scores[i] = Double.MAX_VALUE;
129 					} else {
130 						scores[i] = Double.valueOf(scoreString);
131 					}
132 					if (scores[i] > minScore) interesting = true;
133 				}
134 				if (!interesting) continue;
135 				LOGGER.debug("Score over " + minScore);
136 				
137 				String uniprotAcc = fields[0];
138 				Entry uniprotEntry = mm.getEntryForAccession(
139 						MmDatabase.UniProt, uniprotAcc);
140 				// XXX: Ignore if not found in the enzymes mega-map:
141 				if (uniprotEntry == null) continue;
142 				LOGGER.debug("UniProt accession is an enzyme: " + uniprotAcc);
143 				
144 				Collection<XRef> diseaseXrefs = new ArrayList<XRef>();
145 
146 				String[] omimCell = fields[1].split("\\s");
147 				addOmimXrefs(uniprotEntry, diseaseXrefs, omimCell);
148 				
149 				String[] meshIdsCell = fields[2].split(" ?/ ?");
150 				String[] meshHeadsCell = fields[3].split(" / ");
151 				addMeshAndEfoXrefs(uniprotEntry, diseaseXrefs, scores,
152 						meshIdsCell, meshHeadsCell);
153 				mm.writeXrefs(diseaseXrefs);
154 			}
155             LOGGER.info("Parsing end");
156             mm.closeMap();
157             LOGGER.info("Map closed");
158         } catch (Exception e){
159             LOGGER.error("During parsing", e);
160             mm.handleError();
161             throw e;
162 		} finally {
163 			if (is != null) is.close();
164 			if (br != null) br.close();
165 		}
166 	}

168 	/**
169 	 * Splits the fields in one line of the file, namely:
170 	 * <ul>
171 	 * <li>[0] - UniProt accession</li>
172 	 * <li>[1] - MIM number(s)</li>
173 	 * <li>[2] - MeSH ID(s)</li>
174 	 * <li>[3] - MeSH heading(s)</li>
175 	 * <li>[4] - Score(s)</li>
176 	 * </ul>
177 	 * @param format the {@link Format} of the file.
178 	 * @param line one line read from the file.
179 	 * @return the split fields in the line, or <code>null</code> if it is a
180 	 * 		header line. Note that multi-valued fields must be split further.
181 	 */
182 	private String[] getFields(Format format, String line) {
183 		String[] fields = null;
184 		switch (format) {
185 		case html:
186 			Matcher m = htmlTablePattern.matcher(line);
187 			// Discard header lines:
188 			if (!m.matches()) return null;
189 			fields = new String[5];
190 			fields[0] = m.group(1).replaceAll("<\\/?a[^>]*>", "");
191 			fields[1] = m.group(2).replaceAll("<\\/?a[^>]*>", "");
192 			fields[2] = m.group(3).replaceAll("<\\/?a[^>]*>", "");
193 			fields[3] = m.group(4);
194 			fields[4] = m.group(5);
195 			break;
196 		case tab:
197 			// Discard header lines:
198 			if (line.startsWith("Swiss-Prot")) return null;
199 			fields = line.split("\t");
200 			break;
201 		}
202 		return fields;
203 	}

205 	/**
206 	 * Adds xrefs from a UniProt entry to the passed MeSH terms if their score
207 	 * is not less than the {@link #minScore}, and to EFO entries whose disease
208 	 * name matches exactly the MeSH term.
209 	 * @param uniprotEntry the UniProt entry referencing MeSH
210 	 * @param diseaseXrefs the collection to update with any cross-references
211 	 * 		to MeSH and EFO.
212 	 * @param scores the scores of MeSH terms for the UniProt accession.
213 	 * @param meshIds the MeSH identifiers.
214 	 * @param meshHeads the MeSH headers (terms).
215 	 */
216 	private void addMeshAndEfoXrefs(Entry uniprotEntry,
217 			Collection<XRef> diseaseXrefs, double[] scores,
218 			String[] meshIds, String[] meshHeads) {
219 		// Same number of MeSH IDs, headers and scores
220 		for (int i = 0; i < scores.length; i++) {
221 			if (scores[i] < minScore) continue;
222 			Entry meshEntry = new Entry();
223 			meshEntry.setDbName(MmDatabase.MeSH.name());
224 			final String meshId = meshIds[i].trim();
225 			meshEntry.setEntryId(meshId);
226 			final String meshHead = meshHeads[i].trim();
227 			meshEntry.setEntryName(meshHead);
228 			
229 			XRef meshXref = new XRef();
230 			meshXref.setFromEntry(uniprotEntry);
231 			meshXref.setRelationship(Relationship.between(
232 					MmDatabase.UniProt, MmDatabase.MeSH).name());
233 			meshXref.setToEntry(meshEntry);
234 			diseaseXrefs.add(meshXref);
235 			LOGGER.debug("MeSH xref to " + meshId);
236 
237 			// Now look if the MeSH heading matches an EFO entry:
238 			try {
239 				Disease efoDisease = (Disease) bioportalAdapter.searchConcept(
240 						BioportalOntology.EFO,
241 						meshHead, Disease.class, false);
242 				if (efoDisease != null){
243 					Entry efoEntry = new Entry();
244 					efoEntry.setDbName(MmDatabase.EFO.name());
245 					efoEntry.setEntryId(efoDisease.getId());
246 					efoEntry.setEntryName(efoDisease.getName());
247 
248 					XRef efoXref = new XRef();
249 					efoXref.setFromEntry(uniprotEntry);
250 					efoXref.setRelationship(Relationship.between(
251 							MmDatabase.UniProt, MmDatabase.EFO).name());
252 					efoXref.setToEntry(efoEntry);
253 					diseaseXrefs.add(efoXref);
254 					LOGGER.debug("EFO xref to " + efoDisease.getId());
255 				}
256 			} catch (BioportalAdapterException e) {
257 				// Don't stop the others just because of BioPortal/EFO problems:
258 				LOGGER.error("Unable to get EFO xref for " + meshHead, e);
259 			}
260 		}
261 	}

263 	/**
264 	 * Adds xrefs from a UniProt entry to the passed OMIM identifiers.
265 	 * @param uniprotEntry the UniProt entry referencing OMIM
266 	 * @param diseaseXrefs the collection to update with any cross-references
267 	 * 		to OMIM.
268 	 * @param omimIds the OMIM identifiers.
269 	 * @throws BioportalAdapterException
270 	 */
271 	private void addOmimXrefs(Entry uniprotEntry,
272 			Collection<XRef> diseaseXrefs, String[] omimIds)
273 	throws BioportalAdapterException {
274 		for (int i = 0; i < omimIds.length; i++) {
275 			final String omimString = omimIds[i].trim();
276 			if (omimString.equals("-")) continue;
277 			Entry omimEntry = new Entry();
278 			omimEntry.setDbName(MmDatabase.OMIM.name());
279 			omimEntry.setEntryId(omimString);
280 			// We need a disease name:
281 			Disease omimDisease = (Disease) bioportalAdapter.searchConcept(
282 					BioportalOntology.OMIM, omimString, Disease.class, false);
283 			if (omimDisease != null){
284 				omimEntry.setEntryName(omimDisease.getName());
285 			}
286 			XRef omimXref = new XRef();
287 			omimXref.setFromEntry(uniprotEntry);
288 			omimXref.setRelationship(Relationship.between(
289 					MmDatabase.UniProt, MmDatabase.OMIM).name());
290 			omimXref.setToEntry(omimEntry);
291 			diseaseXrefs.add(omimXref);
292 			LOGGER.debug("OMIM xref to " + omimString);
293 		}
294 	}
