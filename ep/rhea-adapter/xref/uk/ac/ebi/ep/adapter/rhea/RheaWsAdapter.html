39      public ReactionPathway getReactionPathway(Reaction reaction) {
40          List<Object> reactionElements = reaction.getReactiveCentreAndMechanismAndReactantList();
41          ReactionPathway reactionPathway = new ReactionPathway();
42          EnzymeReaction enzymeReaction = new EnzymeReaction();
43  //        reactionElements.get(0);
44          Equation equation = new Equation();
45          equation.setDirection(createEquationSymbol(reaction));
46          List<Link> linkList = new ArrayList<Link>();
47          for (Object reactionElement : reactionElements) {
48              if (reactionElement instanceof Map) {
49                  Map xRefs = (Map) reactionElement;
50                  linkList.addAll(xRefs.getLink());
51              }
52              if (reactionElement instanceof ReactantList) {
53                  List<Object> reactantList = ((ReactantList) reactionElement)
54                  		.getReactantListOrReactant();
55                  equation.getReactantlist().addAll(reactantList);
56              }
57              if (reactionElement instanceof ProductList) {
58                  List<Object> productList = ((ProductList) reactionElement)
59                  		.getProductListOrProduct();
60                  equation.getProductlist().addAll(productList);
61  
62              }
63          }
64          //Set reaction properties
65          String rheaId = reaction.getId();
66          enzymeReaction.setId(rheaId);
67          String reactionTitle = createReactionTitle(equation);
68          enzymeReaction.setName(reactionTitle);
69          enzymeReaction.setEquation(equation);
70          reactionPathway.setReaction(enzymeReaction);
71          List<Object> mecList = getMechanismLinks(linkList);
72          reactionPathway.setMechanism(mecList);
73          List<Object> pwList = getPathwayLinks(linkList);
74          setReactomeReactionRef(reactionPathway,pwList);
75          return reactionPathway;
76      }

100     private String createEquationSymbol(String direction) {
101         String symbol = null;
102         if (direction.equalsIgnoreCase("direction.UN")) {
103             symbol = " <?> ";
104         }
105         if (direction.equalsIgnoreCase("direction.BI")) {
106             symbol = " <=> ";
107         }
108         if (direction.equalsIgnoreCase("direction.LR")) {
109             symbol = " => ";
110         }
111         if (direction.equalsIgnoreCase("direction.RL")) {
112             symbol = " <= ";
113         }
114         return symbol;
115     }

117     private String createReactionTitle(Equation equation) {

127     private String createParticipantString(List<Object> molList) {
128         StringBuffer sb = new StringBuffer();
129          int counter = 1;
130         for (Object participant : molList) {
131         	Molecule molecule = null;
132         	Double coef = null;
133         	if (participant instanceof Reactant){
134         		final Reactant reactant = (Reactant) participant;
135 				molecule = reactant.getMolecule();
136 				coef = reactant.getCount();
137         	} else if (participant instanceof Product){
138         		final Product product = (Product) participant;
139 				molecule = product.getMolecule();
140 				coef = product.getCount();
141         	} else {
142         		throw new IllegalArgumentException("");
143         	}
144         	if (coef > 1){
145         		sb.append(coef.intValue()).append(' ');
146         	}
147             sb.append(molecule.getTitle());
148             if (counter < molList.size()) {
149                 sb.append(" + ");
150             }
151             counter++;
152         }
153         return sb.toString();
154     }
