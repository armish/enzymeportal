57      public Map<String, Set<String>> getSynonyms(Set<String> ecNumbers)
58  	throws MultiThreadingException {
59          ExecutorService pool = Executors.newCachedThreadPool();
60          CompletionService<Set<String>> ecs =
61         		 	new ExecutorCompletionService<Set<String>>(pool);
62          Map<String, Set<String>> resultMap =
63          		new Hashtable<String, Set<String>>();
64          Map<Future<Set<String>>, String> future2ec =
65          		new HashMap<Future<Set<String>>, String>();
66          try {
67  	        for (String ecNumber:ecNumbers) {
68  	        	// Avoid dashes, there are no IntEnzXML files for those:
69  	        	if (ecNumber.indexOf('-') > -1) continue;
70  	            Callable<Set<String>> callable = new GetSynonymsCaller(
71  	                    IntenzUtil.createIntenzEntryUrl(
72  	                    		config.getIntenzXmlUrl(), ecNumber));
73  	            future2ec.put(ecs.submit(callable), ecNumber);
74  	        }
75  	        for (int i = 0; i < future2ec.size(); i++){
76  	        	try {
77  	        		Future<Set<String>> future =
78  	        				ecs.poll(config.getTimeout(), TimeUnit.MILLISECONDS);
79  	        		String ec = future2ec.get(future);
80  	        		if (future != null){
81  			            Set<String> synonyms = future.get();
82  		                if (synonyms.size() > 0) {
83  		                    resultMap.put(ec, synonyms);
84  		                }
85  	        		} else {
86  	        			LOGGER.warn("Job not returned getting synonyms!");
87  	        		}
88  		        } catch (Exception e) {
89  		        	// Don't stop the others
90  		        	LOGGER.error("Getting synonyms", e);
91  	        	}
92  	        }
93  //        	LOGGER.debug("SEARCH synonyms after getting futures");
94          } finally {
95          	pool.shutdown();
96          }
97          return resultMap;
98      }

113                 GetEcHierarchyCaller ecCaller = new GetEcHierarchyCaller();

116 //                LOGGER.debug("SEARCH before getEcHierarchy");
117                 EnzymeHierarchy enzymeHierarchy = ecCaller.getEcHierarchy(intenz);
118                 if (enzymeHierarchy != null) {
119                     enzymeHierarchies.add(enzymeHierarchy);
120                 }
121 //                LOGGER.debug("SEARCH after  getEcHierarchy");
122             }
123 //            LOGGER.debug("SEARCH after intenzList");
124             if (synonyms.size() > 0) {
125             	@SuppressWarnings("unchecked")
126 				Collection<String> newSynonyms = CollectionUtils.subtract(
127 						synonyms, enzymeModel.getSynonym());
128                 enzymeModel.getSynonym().addAll(newSynonyms);
129             }
130 
131             //Enzyme has previously initiatized to set Sequence info
132             //Enzyme enzyme = new Enzyme();
133 
134             if (enzymeModel.getEnzyme()  == null) {
135                 Enzyme enzyme = new Enzyme();
136                 enzymeModel.setEnzyme(enzyme);
137             }
138             if (enzymeHierarchies.size() > 0) {
139                 enzymeModel.getEnzyme().setEchierarchies(enzymeHierarchies);
140             }        
141         }

146      public List<Intenz> getIntenz(Collection<String> ecList) {
147          List<Intenz> results = new ArrayList<Intenz>();
148          ExecutorService pool = Executors.newCachedThreadPool();
149          CompletionService<Intenz> ecs =
150         		 new ExecutorCompletionService<Intenz>(pool);
151          try {
152 //        	 LOGGER.debug("SEARCH before callables loop");
153         	 List<String> incompleteEcs = new ArrayList<String>();
154              for (String ec : ecList) {
155             	 if (ec.indexOf('-') > -1){
156             		 incompleteEcs.add(ec);
157             		 continue; // ignore incomplete ECs
158             	 }
159                 Callable<Intenz> callable = new GetIntenzCaller(
160                 		IntenzUtil.createIntenzEntryUrl(
161                 				config.getIntenzXmlUrl(), ec));
162                 ecs.submit(callable);
163              }
164              if (!incompleteEcs.isEmpty()){
165             	 LOGGER.warn("Unable to retrieve info from IntEnz on " +
166             	 		incompleteEcs);
167              }
168 //        	 LOGGER.debug("SEARCH before futures loop");
169              int completeEcs = ecList.size() - incompleteEcs.size();
170              for (int i = 0; i < completeEcs; i++) {
171             	 try {
172                 	 Future<Intenz> future =
173                 			 ecs.poll(config.getTimeout(), TimeUnit.MILLISECONDS);
174                      if (future != null) {
175                          Intenz intenz = future.get();
176                          if (intenz != null) results.add(intenz);
177                      } else {
178                     	 LOGGER.warn("Job not returned!");
179                      }
180             	 } catch (Exception e){
181             		 // Don't stop the others
182  	            	LOGGER.error("Callable " + (i+1) + " of " + completeEcs
183 	            			+ " - " + e.getMessage(), e);
184             	 }
185              }
186 //        	 LOGGER.debug("SEARCH after  futures loop");
187          } finally {
188         	 pool.shutdown();
189          }
190          return results;
191      }

193 	public Collection<Molecule> getCofactors(String ec) {
194 		// Incomplete EC numbers won't get anything from IntEnzXML:
195 		if (ec.indexOf('-') > -1) return null;
196 		Collection<Molecule> cofactors = null;
197 		GetCofactorsCaller cofactorsCaller = new GetCofactorsCaller(
198 			IntenzUtil.createIntenzEntryUrl(config.getIntenzXmlUrl(), ec));
199 		try {
200 			cofactors = cofactorsCaller.call();
201 		} catch (Exception e) {
202 			LOGGER.error("Unable to retrieve cofactors for " + ec, e);
203 		}
204 		return cofactors;
205 	}

207 	public Collection<Molecule> getCofactors(Collection<String> ecs) {
208 		Collection<Molecule> cofactors = null;
209 		for (String ec : ecs) {
210 			if (ec.indexOf('-') > -1) continue;
211 			final Collection<Molecule> newCofactors = getCofactors(ec);
212 			if (cofactors == null){
213 				cofactors = newCofactors;
214 			} else if (newCofactors != null){
215 				cofactors.addAll(newCofactors);
216 			}
217 		}
218 		return cofactors;
219 	}
