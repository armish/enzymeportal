138     private void processData(Set<CustomEntity> allCompounds) {
139         List<XRef> xRefList = new ArrayList<XRef>();
140         for (CustomEntity customEntity : allCompounds) {
141 
142 
143             chebiEntry = computeChebiEntry(customEntity.getCompounds().getChebiAccession(), customEntity.getCompounds().getName());
144 
145             uniprotEntry = computeUniprotEntry(customEntity.getReference().getReference_id(), customEntity.getReference().getReference_name());
146 
147             xRef = computeXref(chebiEntry, uniprotEntry, customEntity.getRelationship());
148 
149             xRefList.add(xRef);
150 
151 
152         }
153 
154         try {
155             
156             LOGGER.info("About to Write to Mega Mapper Database....");
157             this.writeXrefs(xRefList);
158             LOGGER.info("Done writing.");

164     private String buildRelationship(String relations) {
165 
166 
167         MEGA_RELATIONSHIP mega_relationship = MEGA_RELATIONSHIP.getEnumFromString(relations);
168         Relationship relationship = null;// Relationship.is_related_to;

171         switch (mega_relationship) {
172             case CATALYTIC_ACTIVITY:
173                 relationship = Relationship.is_substrate_or_product_of;
174                 break;
175             case COFACTOR:
176                 relationship = Relationship.is_cofactor_of;
177                 break;
178             case ENZYME_REGULATION:
179                 relationship = Relationship.is_related_to;
180                 break;
181             case INDUCTION:
182                 relationship = Relationship.is_related_to;
183                 break;
184             case PHARMACEUTICAL:
185                 relationship = Relationship.is_related_to;
186                 break;
187             case PTM:
188                 relationship = Relationship.is_related_to;
189                 break;
190             case TOXIC_DOSE:
191                 relationship = Relationship.is_related_to;
192                 break;
