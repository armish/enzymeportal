38  	 * Wrapper around a CiteXplore citation which implements
39  	 * {@link #equals(Object)} and {@link #hashCode()} methods, and

43  	public class LabelledCitation implements Comparable<LabelledCitation> {
44  		private Result citation;

59  		/**
60  		 * Equals method based on bibliography identifier, namely PubMed ID.
61  		 * It does not take labels into account.
62  		 */
63  		@Override
64  		public boolean equals(Object o) {
65  			if (!(o instanceof LabelledCitation)) return false;
66  			if (o == this) return true;
67  			LabelledCitation other = (LabelledCitation) o;
68  			String tcei = this.citation.getId();
69  			String ocei = other.citation.getId();
70  			if (tcei == null ^ ocei == null) return false;
71  			if (tcei != null && ocei != null){
72  				String tcds = this.citation.getSource();
73  				String ocds = other.citation.getSource();
74  				return tcei.equals(ocei) && tcds.equals(ocds);
75  			}
76  			return this.citation.getTitle().equals(other.citation.getTitle());
77  		}
78  		@Override
79  		public int hashCode() {
80  			int hash = 17;
81  			hash = hash * 17;
82  			if (citation.getSource() != null){
83  				hash += citation.getSource().hashCode();
84  			}
85  			hash = hash * 17;
86  			if (citation.getId() != null){
87  				hash += citation.getId().hashCode();
88  			}
89  			hash = hash * 17 + citation.getTitle().hashCode();
90  			return hash;
91  		}
92  		/**
93  		 * By default, citations are sorted by creation date.
94  		 */
95  		public int compareTo(LabelledCitation other) {
96  			if (other == this) return 0;
97  			return this.citation.getDateOfCreation()
98  					.compare(other.citation.getDateOfCreation());
99  		}

105 	public List<LabelledCitation> getCitations(String uniprotId, List<String> pdbIds) {
106 		List<LabelledCitation> citations = null;
107 		LOGGER.debug("Before getting lit. callables");
108 		List<Callable<Set<Result>>> callables = getCallables(uniprotId, pdbIds);
109 		List<Future<Set<Result>>> futures = new ArrayList<Future<Set<Result>>>();
110 		LOGGER.debug("Before creating thread pool");
111 
112 		ExecutorService pool =
113 				Executors.newFixedThreadPool(config.getMaxThreads());
114 		CompletionService<Set<Result>> ecs =
115 				new ExecutorCompletionService<Set<Result>>(pool);
116 		try {
117 			for (Callable<Set<Result>> callable : callables) {
118 				futures.add(ecs.submit(callable));
119 			}
120 			for (int i = 0; i < callables.size(); i++){
121 				Future<Set<Result>> future = null;
122 				try {
123 					future = ecs.take();
124 					if (future == null){
125 						LOGGER.error("No citations retrieved for...");
126 					} else {
127 						Set<Result> cits = future.get();
128 						if (cits != null){
129 							CitationLabel label = getLabel(
130 									callables.get(futures.indexOf(future)));
131 							for (Result cit : cits) {
132 								LabelledCitation citation =
133 										new LabelledCitation(cit, label);
134 								if (citations == null){
135 									citations = new ArrayList<LabelledCitation>();
136 								} else if (citations.contains(citation)){
137 									citations.get(citations.indexOf(citation))
138 											.addLabel(label);
139 								} else {
140 									citations.add(citation);
141 								}
142 							}
143 						}
144 					}
145 				} catch (Exception e) {
146 					LOGGER.error("Unable to send requests for citations", e);
147 				}
148 			}
149 		} finally {
150 			pool.shutdown();
151 		}
152 		LOGGER.debug("After getting citations");
153 		Collections.sort(citations, Collections.reverseOrder());
154 		return citations;
155 	}

157 	private List<Callable<Set<Result>>> getCallables(String uniprotId,
158 			List<String> pdbIds) {
159 		List<Callable<Set<Result>>> callables =
160 				new ArrayList<Callable<Set<Result>>>();
161 		callables.add(new UniprotJapiLiteratureCaller(uniprotId));
162 		for (String pdbId : pdbIds) {
163 			callables.add(new DASLiteratureCaller(
164 					IDASFeaturesAdapter.PDBE_DAS_URL, pdbId));
165 		}
166 		return callables;
167 	}
