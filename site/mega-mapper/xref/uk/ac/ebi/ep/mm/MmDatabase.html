83  	/**
84  	 * Parses a string into a MmDatabase value. This method is more flexible
85  	 * than <code>valueOf</code>, accepting database synonyms (ex. 'TrEMBL'
86  	 * will be parsed as {@link #UniProt}) and ignoring letter case.
87  	 * @param s
88  	 * @return a MmDatabase, or <code>null</code> if no match is found.
89  	 */
90  	public static MmDatabase parse(String s){
91  		MmDatabase result = null;
92  		try {
93  			result = MmDatabase.valueOf(s);
94  		} catch (Exception e){
95  			dbLoop: for (MmDatabase db : values()) {
96  				// Try ignoring case:
97  				if (db.name().equalsIgnoreCase(s)){
98  					result = db;
99  					break dbLoop;
100 				}
101 				// Try to guess from synonyms:
102 				if (db.synonyms != null){
103 					for (String synonym: db.synonyms){
104 						if (synonym.equalsIgnoreCase(s)){
105 							result = db;
106 							break dbLoop;
107 						}
108 					}
109 				}
110 			}
111 		}
112 		if (result == null){
113 			LOGGER.warn("No database found for " + s);
114 		}
115 		return result;
116 	}
